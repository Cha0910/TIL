## 2. 프로그래머스 문제 풀이

### [[1차] 프렌즈4블록](https://school.programmers.co.kr/learn/courses/30/lessons/17679)

```cpp
#include <string>
#include <vector>
#include <set>

using namespace std;

void remove(vector<string>& board, set<pair<int, int>>& removed, int y, int x){
    char block = board[y][x];
    if(block == '0') return;
    
    int dy[4] = {-1, -1, 1, 1};
    int dx[4] = {-1, 1, 1, -1};
    
    for(int i=0; i<4; i++){
        int ny = y+dy[i], nx = x+dx[i];
        if(ny<0 || ny>=board.size() || nx<0 || nx>=board[0].size()) continue;
        if(board[ny][nx] == board[y][x] && board[ny][x] == board[y][x] && board[y][nx] == board[y][x]){
            removed.insert({ny, nx});
            removed.insert({ny, x});
            removed.insert({y, nx});
            removed.insert({y, x});
        }
    }
}

void move(vector<string>& board, int m, int n){
    for(int i=0; i<n; i++){
        vector<char> blocks;
        
        for(int j=m-1; j>=0; j--){
            char block = board[j][i];
            if(block != '0') blocks.push_back(block);
        }
        
        for(int j=0; j<blocks.size(); j++)
            board[m-1-j][i] = blocks[j];
        for(int j=blocks.size(); j<m; j++)
            board[m-1-j][i] = '0';
    }
}

int solution(int m, int n, vector<string> board) {
    int answer = 0;
    set<pair<int, int>> removed;
    
    do{
        removed.clear();
        
        for(int i=0; i<m; i++)
            for(int j=0; j<n; j++)
                remove(board, removed, i, j);
        
        for(const auto& pos : removed)
            board[pos.first][pos.second] = '0';
        
        answer += removed.size();
        move(board, m, n);
    }while(!removed.empty());
        
    return answer;
}
```

* **구현** 문제
* `remove` 함수로 현재 `board`의 상태에서 지울 수 있는 블록을 `removed`에 저장
* 겹치는 블럭이 있을 수 있으므로 `set`에 저장
* `move`함수로 각 열에 존재하는 블록을 아래로 이동
