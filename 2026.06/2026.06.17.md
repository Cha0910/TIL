### [교점에 별 만들기](https://school.programmers.co.kr/learn/courses/30/lessons/87377)

```cpp
#include <string>
#include <vector>
#include <climits>
using namespace std;

vector<string> solution(vector<vector<int>> line) {
    vector<pair<long long, long long>> points;
    long long xmin = LLONG_MAX, xmax = LLONG_MIN;
    long long ymin = LLONG_MAX, ymax = LLONG_MIN;
    
    for(int i=0; i<line.size(); i++){
        for(int j=i+1; j<line.size(); j++){
            long long a = line[i][0], b = line[i][1], e = line[i][2];
            long long c = line[j][0], d = line[j][1], f = line[j][2];
            long long x = b*f - e*d, y = e*c - a*f;
            long long div = a*d - b*c;
            if(div == 0) continue;
            
            if(x % div == 0 && y % div == 0){
                x /= div; xmin = min(xmin, x); xmax = max(xmax, x);
                y /= div; ymin = min(ymin, y); ymax = max(ymax, y);
                points.push_back({x, y});
            }
        }
    }
    
    for(int i=0; i<points.size(); i++){
        points[i].second -= ymin;
        points[i].first -= xmin;
    }
    ymax -= ymin; ymin = 0;
    xmax -= xmin; xmin = 0;
    
    vector<string> answer(ymax+1, string(xmax+1, '.'));
    for(int i=0; i<points.size(); i++){
        int x = points[i].first, y = ymax - points[i].second;
        answer[y][x] = '*';
    }
    
    return answer;
}
```

* **수학**, **구현** 문제
* 직선의 교점을 구해 `points`에 저장하면서 좌표의 최대, 최솟값을 저장
* 이후 최솟값을 `(0, 0)`으로 **평행이동**시켜 그림을 그림

### [상담원 인원](https://school.programmers.co.kr/learn/courses/30/lessons/214288)

```cpp
#include <string>
#include <vector>
#include <queue>

using namespace std;

int getWaitingTime(int k, int n, const vector<vector<int>>& reqs){
    int waitingTime = 0, currTime = 0;
    queue<pair<int, int>> reqsQ;
    priority_queue<int, vector<int>, greater<int>> currQ;
    
    for(const vector<int>& req : reqs)
        if(req[2] == k) reqsQ.push({req[0], req[1]});
    
    while(!reqsQ.empty()){
        if(currQ.size() < n){
            pair<int, int> req = reqsQ.front();
            reqsQ.pop();
            if(currTime > req.first) waitingTime += currTime - req.first;
            else currTime = req.first;
            currQ.push(currTime + req.second);    
        }
        else{
            currTime = currQ.top();
            currQ.pop();
        }
    }  
    return waitingTime;
}

void getCandidates(int n, int start,vector<int>& candidate, vector<vector<int>>& candidates){
    if(n==0){
        candidates.push_back(candidate);
        return;
    }
    
    for(int i=start; i<candidate.size(); i++){
        candidate[i]++;
        getCandidates(n-1, i, candidate, candidates);
        candidate[i]--;
    }
}

int solution(int k, int n, vector<vector<int>> reqs) {
    int answer = 10000000;
    vector<vector<int>> candidates;
    vector<int> candidate(k, 1);
    getCandidates(n-k, 0, candidate, candidates);
    vector<vector<int>> waitingTimes(k+1, vector<int>(n+1, 0));
    
    for(int i=1; i<=k; i++)
        for(int j=1; j<=n; j++)
            waitingTimes[i][j] = getWaitingTime(i, j, reqs);
        
    for(const vector<int>& cand : candidates){
        int sum = 0;
        for(int i=1; i<=k; i++)
            sum += waitingTimes[i][cand[i-1]];
        if(sum < answer) answer = sum;
    }
    
    return answer;
}
```

* **시뮬레이션** 유형의 문제, **조합**, **우선순위큐**를 활용해 풀이
* `getCandidates` 함수를 통해 `n`명의 상담원을 `k`개의 상담 유형에 배치하는 모든 경우를 구함
* `getWaitingTime` 함수로 `k`번째 상담 유형에, `n`명의 상담원이 배치된 경우 **상담자가 기다리는 시간**을 **우선순위큐**를 활용해 시뮬레이션
* 이후 모든 경우에 대해서 최솟값 반환

## 3. LeetCode 문제풀이

### [3Sum](https://leetcode.com/problems/3sum/description/)

```cpp
#include <algorithm>

class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> answer;
        sort(nums.begin(), nums.end());
        
        for(int i=0; i<nums.size(); i++){
            if(nums[i] > 0) break;
            if (i>0 && nums[i] == nums[i-1]) continue;
            int left = i+1, right = nums.size()-1;

            while(left<right){
                int sum = nums[i] + nums[left] + nums[right];
                if (sum > 0) right--;
                else if (sum < 0) left++;
                else {
                    answer.push_back({nums[left], nums[right], nums[i]});
                    while (left<right && nums[left] == nums[left+1]) left++;
                    while (left<right && nums[right] == nums[right-1]) right--;
                    left++;
                    right--;
                }
            }
        }
        return answer;
    }
};
```

* **투 포인터** 유형의 문제
* 3개의 수를 합한 경우를 고려해야 하기에, **수 하나를 고정**한 후 나머지 두 개의 수를 투 포인터로 연산
* 정답에 중복되는 원소가 없어야 하므로 **중복되는 원소는 건너 뜀**

### [Valid Palindrome](https://leetcode.com/problems/valid-palindrome/)

```cpp
class Solution {
public:
    bool isPalindrome(string s) {
        int left = 0, right = s.length()-1;
        while(left<right && !isalpha(s[left]) && !isdigit(s[left])) left++;
        while(left<right && !isalpha(s[right]) && !isdigit(s[right])) right--;

        while(left<right){
            if(tolower(s[left]) != tolower(s[right])) return false;
            left++;
            right--;
            while(left<right && !isalpha(s[left]) && !isdigit(s[left])) left++;
            while(left<right && !isalpha(s[right]) && !isdigit(s[right])) right--;
        }
        return true;
    }
};
```

* **투 포인터** 유형의 문제
* 알파벳, 숫자 외의 문자는 건너뛰어야 하며, 소문자로 전환해 비교

## 4. 언리얼 엔진의 멀티스레드

### Game Thread

* 언리얼 엔진의 **게임의 전반적인 로직**을 담당하는 메인 스레드

* 오브젝트 업데이트, 틱 실행, 입력 처리, GC 등 게임의 중심이 되는 로직이 실행됨

* 언리얼의 UObject 시스템은 **thread-safe 하지 않기 때문에** 다른 스레드에서 접근 시 **크래시**를 발생

* Game Thread 실행하는 무거운 작업을 작은 **Task**로 분할해 실행하여 CPU 멀티코어 활용도를 높일 수 있음

### Render Thread

* **Game Thread**가 계산한 결과를 복제해 **렌더링 명령을 생성**하는 스레드

* 가시성(Culling) 계산, 드로우 콜 조립, 라이팅 계산 등의 작업을 담당

* GameThread와 Render Thread 사이에는 **Render Command Queue**를 두어 큐를 통해 명령을 전달함

* Game Thread의 이전 프레임을 복제하므로 **Game Thread보다 프레임이 뒤처짐**

### RHI Thread

* **Render Hardware Interface Thread**의 약자로 

* 게임이 사용 중인 **그래픽 API**에 따라서 **Command Buffer**를 생성하고 **GPU의 Command Queue에 제출**

* 과거에는 Render Thread에서 RHI Thread의 역할을 담당해 **병목이 발생**했었음
