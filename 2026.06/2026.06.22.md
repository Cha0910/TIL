## 2. 프로그래머스 문제 풀이

### [퍼즐 조각 채우기](https://school.programmers.co.kr/learn/courses/30/lessons/84021)

```cpp
#include <string>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

vector<vector<pair<int, int>>> BFS(const vector<vector<int>>& game_board, int target){
    queue<pair<int, int>> q;
    vector<vector<pair<int, int>>> result;
    vector<vector<bool>> visit(game_board.size(), vector<bool>(game_board[0].size(), false));
    
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    for(int i=0; i<game_board.size(); i++){
        for(int j=0; j<game_board[i].size(); j++){
            if(game_board[i][j] != target || visit[i][j]) continue;
            vector<pair<int, int>> pos;
            
            q.push({i, j});
            visit[i][j] = true;
            pos.push_back({i, j});
            
            while(!q.empty()){
                pair<int, int> curr = q.front();
                q.pop();
                for(int k=0; k<4; k++){
                    int ny = curr.first + dy[k];
                    int nx = curr.second + dx[k];
                    if(ny<0 || ny>=game_board.size() || nx<0 || nx>=game_board[i].size() ||
                      game_board[ny][nx] != target || visit[ny][nx]) continue;
                    q.push({ny, nx});
                    visit[ny][nx] = true;
                    pos.push_back({ny, nx});
                }
            }
            sort(pos.begin(), pos.end());
            result.push_back(pos);
        }
    }
    return result;
}

vector<vector<pair<int, int>>> rotate(vector<pair<int, int>>& pos){
    vector<vector<pair<int, int>>> result(4, vector<pair<int, int>>(pos.size())); // 0, 90, 180, 270
    int y = pos[0].first, x = pos[0].second;
    for(int i=0; i<pos.size(); i++){
        pos[i].first -= y;
        pos[i].second -= x;
        
        result[0][i] = {pos[i].first, pos[i].second};
        result[1][i] = {-pos[i].second, pos[i].first};
        result[2][i] = {-pos[i].first, -pos[i].second};
        result[3][i] = {pos[i].second, -pos[i].first};
    }
    for(int i=0; i<4; i++)
        sort(result[i].begin(), result[i].end());
    
    return result;
}

int solution(vector<vector<int>> game_board, vector<vector<int>> table) {
    int answer = 0;
    vector<vector<pair<int, int>>> boards = BFS(game_board, 0), blocks = BFS(table, 1);
    vector<bool> usedBoard(boards.size(), false), usedBlock(blocks.size(), false);
    
    for(int i=0; i<boards.size(); i++){
        for(int j=0; j<blocks.size(); j++){
            if(boards[i].size() != blocks[j].size() || usedBoard[i] || usedBlock[j]) continue;
            vector<vector<pair<int, int>>> rotates = rotate(blocks[j]);
            
            for(int k=0; k<4; k++){
                int flag = true;
                int dy = boards[i][0].first - rotates[k][0].first;
                int dx = boards[i][0].second - rotates[k][0].second;
                
                for(int l=1; l<boards[i].size(); l++){
                    if(dy != boards[i][l].first - rotates[k][l].first || 
                       dx != boards[i][l].second - rotates[k][l].second){
                        flag = false;
                        break;
                    }
                }
                
                if(flag) {
                    answer += boards[i].size();
                    usedBoard[i] = true;
                    usedBlock[j] = true;
                    break;
                }
            }
        }
    }
    
    return answer;
}
```

* **BFS**, **시뮬레이션** 문제
* 코드의 구조 자체는 공간과 블럭을 찾고, 공간과 블럭을 비교하는 간단한 알고리즘
* 공간과 블럭의 좌표를 정렬한 후, 모든 좌표의 차가 동일하면 일치하는 것

## 3. LeetCode 문제풀이

### [Combination Sum](https://leetcode.com/problems/combination-sum/description/)

```cpp
class Solution {
public:
    void combination(const vector<int>& candidates, int target, int start, int sum, vector<int>& com, vector<vector<int>>& answer){
        if(sum > target) return;
        if(sum == target){
            answer.push_back(com);
            return;
        }

        for(int i=start; i<candidates.size(); i++){
            int count = 0;
            int next = sum;
            while(next < target){
                count++;
                next += candidates[i];
                com.push_back(candidates[i]);
                combination(candidates, target, i+1, next, com, answer);
            }
            for(int j=0; j<count; j++) com.pop_back();
        }
    }

    vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
        vector<vector<int>> answer;
        vector<int> com;
        combination(candidates, target, 0, 0, com, answer);
        return answer;
    }
};
```

* **백트래킹** 문제
* 하나의 원소가 **여러 번 선택**될 수 있다는 특징이 있음

### [Gas Station](https://leetcode.com/problems/gas-station/description/)

```cpp
class Solution {
public:
    int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
        int gasSum = 0, costSum = 0;
        int answer = 0, current = 0;

        for(int i=0; i<gas.size(); i++){
            gasSum += gas[i];
            costSum += cost[i];
        }
        if(costSum > gasSum) return -1;

        for(int i=0; i<gas.size(); i++){
            current += gas[i];
            current -= cost[i];
            if(current < 0){
                current = 0;
                answer = i+1;
                continue;
            }
        }
        return answer;
    }
};
```

* **그리디** 유형의 문제
* 최대 가스량이 필요량보다 적으면 -1 반환
* 음수가 발생하면 음수가 발생한 지점 전의 어떤 지점에서 시작해도 음수가 나올 수밖에 없음
* 따라서 시작점을 음수 다음 지점으로 설정

### [N-Queens](https://leetcode.com/problems/n-queens/)

```cpp
#include <cmath>
class Solution {
public:
    bool isCan(const vector<int>& board, int r, int c){
        for(int i=0; i<r; i++) 
            if(board[i] == c ||r - i == abs(board[i] - c)) return false;
        return true;
    }

    void NQueens(vector<int>& board, int n, int r, vector<vector<int>>& answer){
        if(r == n) {
            answer.push_back(board);
            return;
        }
        for(int i=0; i<n; i++){
            if(isCan(board, r, i)){
                board[r] = i;
                NQueens(board, n, r+1, answer);
            }
        }
    }

    vector<vector<string>> solveNQueens(int n) {
        vector<vector<int>> boards;
        vector<int> board(n, -1);
        NQueens(board, n, 0, boards);
        vector<vector<string>> answer;
        
        for(int i=0; i<boards.size(); i++){
            vector<string> result;
            for(int j=0; j<n; j++){
                string temp(n, '.');
                temp[boards[i][j]] = 'Q';
                result.push_back(temp);
            }
            answer.push_back(result);
        }
        return answer;
    }
};
```

* **백트래킹** 문제
* 백트래킹의 가장 유명한 문제 중 하나인 N-Queens 문제
* 퀸의 위치를 1차원 벡터로 표현해 검증

### [Assign Cookies](https://leetcode.com/problems/assign-cookies/description/)

```cpp
#include <algorithm>
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int answer = 0, cur = 0;

        for(int i=0; i<g.size(); i++){
            while(cur < s.size() && s[cur] < g[i]) cur++;
            if(cur >= s.size()) return answer;
            cur++;
            answer++;
        }
        return answer;
    }
};
```

* **그리디**, **투 포인터** 유형의 문제
* 아이들에게 줄 수 있는 가장 작은 쿠키를 줌

### [Minimum Path Sum](https://leetcode.com/problems/minimum-path-sum/description/)

```cpp
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        for(int i=1; i<grid.size(); i++) grid[i][0] += grid[i-1][0];
        for(int i=1; i<grid[0].size(); i++) grid[0][i] += grid[0][i-1];

        for(int i=1; i<grid.size(); i++)
            for(int j=1; j<grid[i].size(); j++)
                grid[i][j] += min(grid[i-1][j], grid[i][j-1]);
            
        return grid[grid.size()-1][grid[0].size()-1];
    }
};
```

* **DP** 문제
* 1열, 1행은 누적 합으로 저장
* 이후 왼쪽, 위의 값 중 낮은 값을 선택

### [House Robber](https://leetcode.com/problems/house-robber/description/)

```cpp
class Solution {
public:
    int rob(vector<int>& nums) {
        if(nums.size() < 2) return nums[0];
        vector<int> dp(nums.size(), 0);
        dp[0] = nums[0];
        dp[1] = max(dp[0], nums[1]);
        for(int i=2; i<nums.size(); i++)
            dp[i] = max(dp[i-1], dp[i-2] + nums[i]);
        return dp[nums.size()-1];
    }
};
```

* **DP** 문제
* 유명한 도둑 DP 문제
* i번째 집을 턴 경우, 안 턴 경우 중 최댓값을 저장

### [Partition Equal Subset Sum](https://leetcode.com/problems/partition-equal-subset-sum/description/)

```cpp
class Solution {
public:
    bool canPartition(vector<int>& nums) {
        int sum = 0;
        for(int i=0; i<nums.size(); i++) sum += nums[i];
        if(sum % 2 != 0) return false;

        vector<bool> dp(sum/2+1, false);
        int target = sum/2;
        dp[0] = true;

        for(int i=0; i<nums.size(); i++){
            for(int j=target; j>=nums[i]; j--)
                if(dp[j - nums[i]]) dp[j] = true;
            if(dp[target]) return true;
        }
        return false;
    }
};
```

* **DP** 문제
* `dp[i]` 는 부분 집합으로 `i`를 만들 수 있다는 의미

## 4. 매크로 함수와 인라인 함수

### 매크로

* 코드를 컴파일하기 전, **전처리기**에 의해 특정 코드를 **치환**하는 기능

* `#define [상수이름] [실제값]` 지시어를 통해 정의할 수 있음
    ```cpp
    #include <iostream>
    #define PI (3.14)

    using namespace std;

    int main() {
        double radius = 5.0; 
        double area = PI * radius * radius; // 3.14 * radius * radius 로 치환

        cout << "반지름: " << radius << endl;
        cout << "원의 넓이: " << area << endl;

        // Output:
        // 반지름: 5
        // 원의 넓이: 78.5

        return 0;
    }
    ```

* **장점**
    * 코드의 **가독성**을 향상할 수 있음
    * 매크로 수정만으로 전체 코드를 수정해 **유지보수성**을 향상할 수 있음
    * 변수가 아닌 단순 **치환**이므로 **메모리, 성능 소모가 없음**

* **단점**
    * 타입을 검사하지 않아 **타입 안정성**이 떨어짐
    * **디버그** 시 매크로 이름으로 보이지 않고 설정한 값으로 보임
    * 스코프 제한이 없어 **모든 파일에 영향**을 줌
    * 단순 치환으로 인한 **의도치 않은 결과**가 나올 수 있음

### 매크로 함수

* 상수 정의 매크로와 동일하지만, **매개변수**를 받아 마치 **함수처럼 치환**해 주는 매크로

* `#define [함수이름](매개변수) [코드]` 지시어를 통해 정의할 수 있음

* 상수 매크로와 **장단점을 공유**함
    ```cpp
    #include <iostream>
    #define SQUARE(x) x * x

    using namespace std;

    int main() {
        int result = SQUARE(5); // 5 * 5 로 치환    
        cout << "5의 제곱: " << result << endl; // Output: 25

        int bad_result = SQUARE(2 + 3); // 2 + 3 * 2 + 3 으로 치환
        cout << "2 + 3의 제곱: " << bad_result << endl; // Output: 11
        
        return 0;
    }
    ```

* 이러한 단점 때문에 현대의 C++에서는 **매크로 사용을 지양**함

### 인라인 함수

* **함수 호출 오버헤드**를 줄이기 위한 기능

* 매크로 함수의 장점을 유지하고, 단점을 해결

* 매크로 함수와 유사하게 함수 호출 시 **복사하여 호출**하는 방식

* 매크로 함수와 달리, 전처리기가 아닌 **컴파일러가 관리**하기에 **반드시 헤더 파일에 작성**해야 함

* 컴파일러는 짧은 함수는 **자동으로 인라인화** 해주거나, 인라인화 할 시 성능상 손해를 보는 경우 **자동으로 무시**함

* **장점**
    * 타입 검사를 통한 **타입 안정성**을 챙김
    * **디버깅**이 가능함
    * **함수 호출 오버헤드**를 감소시킬 수 있음
    * 함수 중복 정의로 인한 **링커 에러를 방지**할 수 있음

    ```cpp
    #include <iostream>

    using namespace std;

    inline int square(int x) {
        return x * x;
    }

    int main() {
        int result = square(5); 
        cout << "5의 제곱: " << result << endl; // 출력: 25


        int safe_result = square(2 + 3); 
        cout << "2 + 3의 제곱: " << safe_result << endl; // 출력: 25

        // 타입 검사로 컴파일 에러
        // square("hello"); 

        return 0;
    }
    ```
