### [k진수에서 소수 개수 구하기](https://school.programmers.co.kr/learn/courses/30/lessons/92335)

```cpp
#include <string>
#include <vector>
#include <cmath>
#include <sstream>
#include <algorithm>
using namespace std;

// 소수 판별
bool isPrime(long long n){
    if(n == 0 || n == 1) return false;
    for(int i=2; i<=sqrt(n); i++)
        if(n % i == 0) return false;
    return true;
}

// k진법 변환
string convertK(int n, int k){
    string result;
    while(n >= k){
        result += to_string(n % k);
        n /= k;
    }
    result += to_string(n);
    reverse(result.begin(), result.end());
    return result;
}

// k 진법의 수를 long long 타입의 벡터로 나눔
vector<long long> split(string str, char c){
    vector<long long> result;
    stringstream ss(str);
    string temp;
    while(getline(ss, temp, c)){
        if(temp.length() > 0)   // 0이 연속으로 있는 경우 제외
            result.push_back(stoll(temp));
    }
    return result;
}

int solution(int n, int k) {
    int answer = 0;
    vector<long long> nums = split(convertK(n, k), '0');
    for(const long long& num : nums) 
        if(isPrime(num)) answer++;
    return answer;
}
```

* `구현` 문제
* 문제의 조건을 구현하기만 하면 되는 문제
* 입력받은 수를 **k진법**의 문자열로 변환한 후
* 0을 기준으로 `split`하면 문제에서 원하는 숫자가 됨
* `split`의 결과를 소수 판별

### [외벽 점검](https://school.programmers.co.kr/learn/courses/30/lessons/60062)

```cpp
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

// 어떤 친구를 시작으로 할지 순열
vector<vector<int>> permutation(const vector<int>& vec, int n){
    vector<vector<int>> result;
    vector<int> temp;
    temp.insert(temp.end(), vec.begin(), vec.begin()+n);
    sort(temp.begin(), temp.end());
    do{
        result.push_back(temp);
    }while(next_permutation(temp.begin(), temp.end()));
    return result;
}
   
// dist.size() 만큼의 친구 수로 모든 외벽을 점검할 수 있는지 확인
bool check(const vector<int>& weak, const vector<int>& dist){
    for(int i=0; i<weak.size()/2; i++){ // 각 취약 지점을 시작점으로
        int count = 0;                  // 커버 되는 취약 지점 수
        int cur = weak[i];              // 시작점
        for(int j=0; j<dist.size(); j++){
            cur += dist[j];             // 현재 친구가 커버할 수 있는 범위 끝
            while(weak[i+count] <= cur) count++;
            cur = weak[i+count];        // 커버되지 않는 취약 지점 중 가장 앞
        }
        if(count == weak.size()/2) return true;
    }
    return false;
}

int solution(int n, vector<int> weak, vector<int> dist) {
    int answer = -1;
    int numWeaks = weak.size();
    sort(dist.rbegin(), dist.rend());
    
    // 원형 배열을 확장해 일자로 바꿈
    for(int i=0; i<numWeaks; i++)
        weak.push_back(weak[i] + n);
    
    for(int i=1; i<=dist.size(); i++){
        vector<vector<int>> per = permutation(dist, i);
        for(int j=0; j<per.size(); j++){
            if(check(weak, per[j])) return i;
        }
    }
    
    return answer;
}
```

* `완전 탐색` 문제
* **원형 배열을 2배**로 늘려 어떤 지점에서 출발했을 때, 직선거리로 계산해도 가능하게 바꿈
* 탐색을 위한 친구를 추가할 때, 거리가 높은 친구를 넣는 것이 효율적이므로 `dist` 를 **내림차순** 정렬
* 각 취약 지점을 시작점으로 친구를 하나씩 투입해 모든 취약 지점을 커버할 수 있는지 확인 
* 친구가 투입되는 순서에 따라서 커버하는 취약 지점이 달라지므로 순열로 모든 경우를 확인

### [지형 편집](https://school.programmers.co.kr/learn/courses/30/lessons/12984)

```cpp
#include <vector>
#include <algorithm>

using namespace std;

long long solution(vector<vector<int>> land, int P, int Q) {
    long long answer = 0;
    vector<pair<long long, long long>> vec;
    vector<int> sortedLand;
    sortedLand.reserve(land.size() * land.size());
    
    // land를 1차원 정렬로 펼친 후 오름차순 정렬
    for(int i=0; i<land.size(); i++)
        sortedLand.insert(sortedLand.end(), land[i].begin(), land[i].end());
    sort(sortedLand.begin(), sortedLand.end());
    
    int cur = -1;
    int currentHeight = -1;
    long long prevWeight = 0;
    
    // 정렬된 land를 돌며 높이와, 빈도 수를 저장
    for(int i=0; i<sortedLand.size(); i++){
        if(currentHeight != sortedLand[i]){
            vec.push_back({sortedLand[i], 1});
            cur++;
        }
        else vec[cur].second++;
        prevWeight += sortedLand[i] * Q;    // 높이를 0으로 할 때의 비용(모두 삭제)
    }
    
    long long size = sortedLand.size();
    answer = prevWeight;
    int prevHeight = 0;
    long long prevSize =0;
    
    // 높이를 점차 올리면서 비용을 계산
    for(const pair<long long, long long>& data : vec){
        long long weight = prevWeight                               // 이전 높이 에서의 비용
            + (prevSize * P * (data.first - prevHeight))            // 새로운 블럭 추가
            - ((size - prevSize) * Q * (data.first - prevHeight));  // 삭제한 블럭 복구

        if(weight < answer) answer = weight;
        
        prevSize += data.second;
        prevWeight = weight;
        prevHeight = data.first;
    }
    
    return answer;
}
```

* `누적합`과 `그리디` 방식으로 해결
* 모든 블럭을 제거했을 때(높이가 0일 때)를 기준으로 비용을 구함
* 그 후, 높이를 점차 올려가며 이전 비용에 추가된 블럭, 삭제된 블럭을 계산하여 비용을 업데이트
* 높이에 따라서 비용은 이차함수 그래프를 그리므로, 비용이 높아지는 순간 멈춤
* 효율성 테스트가 굉장히 빡빡해서 알고리즘을 여러번 수정함
* 초기에는 `해시맵`을 활용하여 `완전 탐색`, 이후 `이진 탐색`, `누적합, 그리디`로 수정해 갔지만 모두 시간초과
* land를 순회하며 `해시맵`의 데이터를 업데이트하고, 이후 `해시맵`을 순회하며 비용을 업데이트하는 것이 시간이 오래 걸린다고 판단해 `해시맵`을 `vector`, `pair`로 바꾸니 효율성 테스트도 통과
* 아마 `이진 탐색`코드도 `vector`, `pair`로 구현했으면 통과 되었을 듯

## 3. UE5 프로젝트 구조

### 프로젝트 구조

#### 프로젝트 주요 파일
* `.sln:` Visual Studio에서 프로젝트를 열기 위한 솔루션 파일, C++ 소스 코드와 빌드 설정이 연결된 핵심 파일
* `.uproject:` 엔진 버전, 포함된 플러그인, 프로젝트 모듈 등의 정보를 담고 있는 JSON 형식의 파일, 더블 클릭 시 언리얼 에디터에서 실행 가능
* `.vsconfig:` Visual Studio Configuration 파일

#### 프로젝트 주요 폴더
* `.vs:` Visual Studio가 자동으로 생성하는 폴더이며, 주로 디버깅과 솔루션 설정에 관련된 파일들이 저장됨
* `Binaries:` 컴파일 결과물(실행 파일, DLL)이 저장됨
* `Config:` 프로젝트의 설정 값(.ini)들이 저장됨. 입력 세팅, 게임 규칙, 엔진 렌더링 설정 등을 관리
* `Content:` 언리얼 에디터의 Content Browser와 연결되어 있으며, 프로젝트의 모든 에셋(메시, 텍스처, 블루프린트, 레벨 등)이 저장됨
* `DerivedDataCache:` 에셋, 쉐이더 등을 빠르게 처리하기 위한 캐시 파일이 저장됨
* `Intermediate:` C++ 컴파일 과정에서 생성되는 임시 파일 및 엔진이 생성한 헤더(Generated.h) 등이 저장됨
* `Saved:` 자동 저장 파일, 로그 (.log), 크래시 덤프 등이 저장됨
* `Source:` 프로젝트의 모든 C++ 소스 코드(.cpp, .h)와 빌드 규칙 파일(.Build.cs, .Target.cs)이 저장됨


### 솔루션 구조

#### 솔루션 주요 폴더
* `Engine:` 언리얼 엔진 자체의 소스 코드와 구성 파일들이 포함된 폴더
* `Games:` 현재 개발 중인 프로젝트의 소스 코드가 포함된 폴더
* `Programs:` 엔진 동작에 필요한 유틸리티 프로그램이나 서버 모듈 (UnrealHeaderTool, UnrealBuildTool 등)
* `Rules:` 프로젝트 빌드 방식과 대상을 정의하는 빌드 규칙 파일(.target.cs, .modules 등)이 포함된 폴더
* `Visualizers:` 디버깅 시 FString, TArray, FName과 같은 언리얼 전용 데이터 타입을 읽기 쉽게 해주는 설정 파일이 포함

#### 프로젝트 루트 폴더 (Games/프로젝트명)
* `Config`, `Source`, `.uproject` 가 있으며 프로젝트 주요 폴더와 동일함
