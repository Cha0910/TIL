### [모음사전](https://school.programmers.co.kr/learn/courses/30/lessons/84512)

```cpp
#include <string>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

int solution(string word) {
    int answer = 0;
    set<string> strings;
    vector<string> chars = {"A", "E", "I", "O", "U", ""};
    for(string c1 : chars){
        for(string c2 : chars){
            for(string c3 : chars){
                for(string c4 : chars){
                    for(string c5 : chars){
                        string str = c1 + c2 + c3 + c4 + c5;
                        strings.insert(str);
                    }
                }
            }   
        }
    }
    auto iter = find(strings.begin(), strings.end(), word);
    answer = distance(strings.begin(), iter);
    return answer;
}
```

* **완전탐색** 문제
* 가능한 문자가 5개밖에 없으므로 만들 수 있는 단어의 수는 `5^6`개 밖에 없음
* 5중 중첩문으로 모든 단어를 담는 사전을 정렬을 위해 `set`으로 만듦
* 찾는 단어가 `set`의 몇 번째 자리에 있는지 반환하면 정답

### [고고학 최고의 발견](https://school.programmers.co.kr/learn/courses/30/lessons/131702)

```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

// 1행을 돌리는 모든 경우의 수
vector<vector<int>> getCandidates(int size, int index, vector<int> candidate){
    vector<vector<int>> result;
    if(index == size) return {candidate};
    
    for(int i=0; i<4; i++){
        candidate[index] = i;
        vector<vector<int>> temp = getCandidates(size, index+1, candidate);
        result.insert(result.end(), temp.begin(), temp.end());
    }
    
    return result;
}

// [y][x] 칸을 돌림
void rotate(vector<vector<int>>& clockHands, int y, int x){
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    clockHands[y][x]++;
    clockHands[y][x] %= 4;
    
    for(int i=0; i<4; i++){
        int nextX = x + dx[i];
        int nextY = y + dy[i];
        if(nextX < 0 || nextX >= clockHands.size() || nextY < 0 || nextY >= clockHands.size())
            continue;
        clockHands[nextY][nextX]++;
        clockHands[nextY][nextX] %= 4;
    }
}

// 1행이 candidate로 고정되었을 때, 모든 칸을 맞추는데 들어가는 비용
int getCount(vector<vector<int>> clockHands, vector<int> candidate){
    int count = 0;
    
    // 1행을 candidate만큼 돌림
    for(int i=0; i<candidate.size(); i++){
        for(int j=0; j<candidate[i]; j++){
            rotate(clockHands, 0, i);
            count++;
        }
    }
    
    // 아래행들을 위 행을 보고 맞춤
    for(int i=1; i<clockHands.size(); i++){
        for(int j=0; j<clockHands.size(); j++){
            for(int k=clockHands[i-1][j]; k%4!=0; k++){
                rotate(clockHands, i, j);
                count++;
            }
        }
    }
    
    // 마지막 행 중 하나라도 0이 아니면 맞춰지지 않은거임
    for(int i=0; i<clockHands.size(); i++){
        if(clockHands[clockHands.size()-1][i] != 0)
            count = INT_MAX;
    }
    
    return count;
}

int solution(vector<vector<int>> clockHands) {
    int answer = INT_MAX;
    vector<vector<int>> candidates = 
        getCandidates(clockHands.size(), 0, vector<int>(clockHands.size(), 0));
    
    for(const vector<int>& candidate : candidates){
        int count = getCount(clockHands, candidate);
        answer = count < answer ? count : answer;
    }
    return answer;
}
```

* **1행의 상태를 고정** 시키면, **2행부터 n행 까지의 회전은 결정** 된다는 것이 문제의 핵심
* 이전에 풀었던 [2차원 동전 뒤집기](https://school.programmers.co.kr/learn/courses/30/lessons/131703)와 비슷한 개념
* `i`행의 상태가 고정되면, `i`행의 칸을 돌리기 위해서는 `i-1`행의 칸을 돌리는 수 밖에 없음
* 이를 `n`행까지 반복하여 `n`행의 상태를 확인한 후, 모두 위를 바라보고 있으면 최소 횟수를 비교해 저장

### [베스트앨범](https://school.programmers.co.kr/learn/courses/30/lessons/42579)

```cpp
#include <string>
#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;

vector<int> solution(vector<string> genres, vector<int> plays) {
    vector<int> answer;
    vector<pair<int, string>> rank;                             // 가장 많이 재생된 장르 순으로 정렬하기 위한 벡터
    unordered_map<string, int> list;                            // 각 장르의 재생 횟수를 더해 저장
    unordered_map<string, vector<pair<int, int>>> genreRank;    // 장르별 랭킹을 저장
    
    for(int i=0; i<genres.size(); i++){
        list[genres[i]] += plays[i];
        genreRank[genres[i]].push_back({plays[i], i});
    }
    
    
    for(auto [genre, play] : list){
        // 각 장르 별 가장 많이 재생된 노래를 정렬
        sort(genreRank[genre].begin(), genreRank[genre].end(),  
             [](const pair<int, int>& a, const pair<int, int>& b){
                 if(a.first != b.first) return a.first > b.first;
                 return a.second < b.second;
             });
        rank.push_back({play, genre});
    }
       
    // 장르의 플레이 수를 합한 결과가 가장 많이 재생된 장르가 앞으로 오게 정렬
    sort(rank.rbegin(), rank.rend());
    
    // 가장 많이 재생된 장르 순으로 노래를 최대 2개씩 저장
    for(int i=0; i<rank.size(); i++){
        vector<pair<int, int>> vec = genreRank[rank[i].second];
        
        for(int j=0; j<2; j++)
            if(j<vec.size()) answer.push_back(vec[j].second);
    }
    
    return answer;
}
```

* `해시`와 `정렬`을 이용해 해결하는 문제
* 자료구조를 잘 이용해서 문제에서 요구하는 조건대로 정렬하기만 하면 됨

### [무지의 먹방 라이브](https://school.programmers.co.kr/learn/courses/30/lessons/42891)

```cpp
#include <string>
#include <vector>
#include <map>

using namespace std;

int solution(vector<int> food_times, long long k) {
    int answer = 0;
    map<long long, int> count;
    int length = food_times.size();
    long long eat_food = 0;
    int target = -1;
    
    // food_time 만큼의 시간이 걸리는 음식의 개수를 오름차 순으로 저장
    for(const int food_time : food_times)
        count[food_time]++;
    
    // 시간이 적게 걸리는 음식부터 제거
    for(auto [food_time, count] : count){
        long long remain_time = food_time - eat_food;   // 가장 낮은 남은 음식을 먹는데 걸리는 시간
        
        if(remain_time*length <= k){    // 현재 남은 시간으로 가장 낮은 시간이 걸리는 음식을 모두 먹을 수 있으면
            k -= remain_time*length;    // 음식을 먹는 데 걸린 시간만큼 남은 시간을 감소
            length -= count;            // 다 먹은 음식은 회전판에서 뺌
            eat_food += remain_time;    // 지금까지 먹은 음식량 증가
        }
        else{                           // 먹을 수 없다는 것은
            target = food_time;         // food_time을 갖고 있는 음식을 먹을 때 방송이 끊긴다는 뜻
            break;
        }
    }
    
    if(target == -1) return target;     // target이 없으면, 모든 음식을 먹을 수 있음
    
    k %= length;    // 남은 시간 동안 두번째 이상의 사이클에서 방송이 터지는 경우를 고려하여 나머지 연산
    for(int i=0; i<food_times.size(); i++){
        if(food_times[i] >= target) k--;    // 다 먹지 못한 음식만 카운트
        if(k < 0) {                         // 시간이 다 되었으면 
            answer = i+1;                   // 그 다음 음식부터 먹기 시작
            break;
        }
    }
    return answer;
}
```

* `그리디` 유형의 문제지만 아이디어를 생각해 내는 게 어려움
* 어떤 숫자의 음식을 먹을 때, 방송이 터지는지 알아내야 함
* 낮은 수의 음식들을 전부 없애는데 얼마나 시간이 걸리는지 파악
* 남은 시간으로 해당 음식을 전부 없애지 못한다면, 그 음식을 먹을 때 방송이 터진다는 것
* 그 숫자의 음식 중 몇 번째 음식을 먹을때 방송이 터지는지 파악하면 정답을 구할 수 있음

### [보석 쇼핑](https://school.programmers.co.kr/learn/courses/30/lessons/67258)

```cpp
#include <string>
#include <vector>
#include <unordered_map>

using namespace std;

vector<int> solution(vector<string> gems) {
    vector<int> answer;
    unordered_map<string, int> count;
    int left = 0;
    int min = gems.size();
    
    // 보석의 종류 개수를 저장
    for(const string& gem : gems) count[gem] = 0;
    int numGems = count.size();
    int num = 0;
    
    // 포인터를 오른쪽으로 한칸 씩 옮김
    for(int right=0; right<gems.size(); right++){
        string gem = gems[right];
        if(count[gem] == 0) num++;  // 처음 추가된 보석이면 종류 개수를 증가
        count[gem]++;               // 현재 구간에서의 보석 개수를 증가
        
        while(count[gems[left]] > 1) {  // 가장 왼쪽에 있는 보석들을 뺄 수 있으면 뺌
            count[gems[left]]--;
            left++;
        }
        
        // 현재 구간에 모든 보석의 종류가 있으면 최솟값을 비교
        if(num == count.size()){
            if(right - left < min){
                min = right - left;
                answer = {left+1, right+1};
            }
        }
    }
        
    return answer;
}
```

* `투 포인터`를 활용하여 해결한 문제
* 오른쪽 보석을 구간에 추가했을 때, 왼쪽 끝에 있는 보석들을 구간에서 제거
* `O(n)` 시간 만에 조건을 만족하는 모든 구간을 구할 수 있음
