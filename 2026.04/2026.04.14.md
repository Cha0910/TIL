### [행렬의 곱셈](https://school.programmers.co.kr/learn/courses/30/lessons/12949)

```cpp
#include <string>
#include <vector>

using namespace std;

vector<vector<int>> solution(vector<vector<int>> arr1, vector<vector<int>> arr2) {
    vector<vector<int>> answer;
    for(int i=0; i<arr1.size(); i++){
        vector<int> temp;
        for(int j=0; j<arr2[0].size(); j++){
            int sum = 0;
            for(int k=0; k<arr2.size(); k++){
                sum += arr1[i][k] * arr2[k][j];
            }
            temp.push_back(sum);
        }
        answer.push_back(temp);
    }
    return answer;
}
```

### [트리 트리오 중간값](https://school.programmers.co.kr/learn/courses/30/lessons/68937)

```cpp
#include <string>
#include <vector>
#include <queue>

using namespace std;

// root 기준 bfs 하여 각 노드의 거리룰 반환
vector<int> bfs(const vector<vector<int>>& graph, const int root){
    vector<int> depth(graph.size(), -1);
    queue<int> q;
    q.push(root);
    depth[root] = 0;
    
    while(!q.empty()){
        int current = q.front();
        q.pop();
        
        for(int i=0; i<graph[current].size(); i++){
            int next = graph[current][i];
            if(depth[next] == -1){
                q.push(next);
                depth[next] = depth[current] + 1;
            }
        }
    }
    return depth;
}

// 최대 거리와 연결된 노드를 반환 index0 : 최대 거리, index1~ : 연결된 노드
vector<int> getMax(const vector<int>& depth){
    int max = -1;
    vector<int> result;
    for(int i=0; i<depth.size(); i++) {
        if(max < depth[i]) {
            max = depth[i];
            result.clear();
            result.push_back(max);
        }
        if(max == depth[i]) 
            result.push_back(i);
    }
    return result;
}

int solution(int n, vector<vector<int>> edges) {
    int answer = 0;
    int root;
    vector<vector<int>> graph(n);
    
    // 그래프 생성
    for(const vector<int>& edge : edges){
        graph[edge[0]-1].push_back(edge[1]-1);
        graph[edge[1]-1].push_back(edge[0]-1);
    }
    // 임의의 리프 노드 선택
    for(int i=0; i<graph.size(); i++){
        if(graph[i].size() == 1) {
            root = i;
            break;
        }
    }
    
    // 임의의 리프 노드에서 최대 거리를 갖는 노드를 선택
    vector<int> depth = bfs(graph, root);
    int leaf = getMax(depth)[1];
    
    // 해당 노드에 대해서 bfs를 실행해 트리의 최대 거리를 구함
    depth = bfs(graph, leaf);
    vector<int> count = getMax(depth);
    
    // 최대 거리를 갖는 노드가 2개 이상이면 최댓값이 정답
    if(count.size() > 2) return count[0];
    
    // 1개면 해당 노드에 대해서 다시 bfs를 실행
    else{
        leaf = count[1];
        depth = bfs(graph, leaf);
        
        // 최대 거리를 갖는 노드가 2개 이상이면 최댓값이 정답
        count = getMax(depth);
        if(count.size() > 2) return count[0];
    }

    // 두 번째 노드도 최대 거리가 한개밖에 없으면 최댓값 - 1이 정답
    answer = count[0]-1;
    return answer;
}
```

* `트리`를 활용하여 푸는 문제 
* 문제를 풀기 위해 가장 중요한 정보는 평균이 아닌 **중간값**이라는 것
* 최솟값은 1을 넣으면 되므로 신경 쓰지 않아도 됨
* 가장 큰 중간값을 구하는 것이므로 **가장 큰 중간값은 최댓값과 똑같거나, 최댓값 - 1** 일 수밖에 없음
* 즉, **트리에서 가장 큰 깊이**를 구한 후 동일한 깊이가 2개인지 확인
* 만약 2개면 중간값은 최댓값과 같아지고, 하나밖에 없으면 최댓값 - 1이 됨

### [경주로 건설](https://school.programmers.co.kr/learn/courses/30/lessons/67259)

```cpp
#include <string>
#include <vector>
#include <queue>

using namespace std;

auto cmp = [](const vector<int>& A, const vector<int>& B){
    return A[0] > B[0];
};

int dijkstra(vector<vector<int>> board, int startY, int startX){
    priority_queue<vector<int>, vector<vector<int>>, decltype(cmp)> pq(cmp);
    int dx[4] = {0, 0, -1, 1};
    int dy[4] = {-1, 1, 0, 0};
    
    pq.push({0, startY, startX, 0});
    pq.push({0, startY, startX, 1});
    
    while(!pq.empty()){
        vector<int> current = pq.top();
        pq.pop();
        int currentY = current[1], currentX = current[2];
        int weight = current[0];
        bool direction = current[3]; // true: 가로 false: 세로
        
        for(int i=0; i<4; i++){
            int y = currentY + dy[i];
            int x = currentX + dx[i];
            int cost = weight + 100;
            if(x<0 || x>=board.size() || y<0 || y>=board.size() || board[y][x] == 1) continue;
            
            // 가로 도로에서 코너
            if(direction && y != currentY) cost += 500;
            // 세로 도로에서 코너
            else if(!direction && x != currentX) cost += 500;
            
            if(board[y][x] == 0 || board[y][x] > cost){
                board[y][x] = cost;
                int dir = cost - weight == 600 ? !direction : direction;
                pq.push({cost, y, x, dir});
            }
        }
    }
    return max(board[0][0], board[board.size()-1][board.size()-1]);
}

int solution(vector<vector<int>> board) {
    int answer = 0;
    answer = min(dijkstra(board, 0, 0), dijkstra(board, board.size()-1, board.size()-1));
    return answer;
}
```

* **다익스트라**와 **DP** 의 개념을 이용하여 해결
* 이전 도로의 방향으로 방향을 꺾었는지 여부를 확인하여 board에 현재까지의 최솟값을 저장
* board는 도로의 방향까지는 저장하지 못하므로 유실이 생길 수밖에 없음
* 도로가 세로인 경우, 가로인 경우를 나누어 저장하는 것이 더 좋겠지만, 너무 늦게 깨달아버림
* 그냥 시작지, 도착지에서 각각 실행해서 유실되는 경우를 찾는 방식으로 정답을 맞추긴 함
* 3차원 배열 쓰는 방식으로 수정해 봐야 할 듯
