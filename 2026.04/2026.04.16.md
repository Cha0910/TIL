### [H-index](https://school.programmers.co.kr/learn/courses/30/lessons/42747)

```cpp
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<int> citations) {
    int answer = 0;
    vector<int> pref(*max_element(citations.begin(), citations.end())+1, 0);
    
    for(const int citation : citations) pref[citation]++;
    for(int i=pref.size()-2; i>=0; i--) pref[i] += pref[i+1];
    for(int i=pref.size()-1; i>=0; i--) if(pref[i] >= i) return i;
    
    return answer;
}
```

* **누적합**을 이용해 해결하는 문제
* `pref[i]`는 `i`번 이상 인용된 논문의 개수를 저장하고 있음
* `i`를 역순으로 순회해 `pref[i] >= i`가 만족하는 처음의 `i`가 **H-index** 점수임

### [동굴 탐험](https://school.programmers.co.kr/learn/courses/30/lessons/67260)

```cpp
#include <string>
#include <vector>
#include <queue>

using namespace std;

bool solution(int n, vector<vector<int>> path, vector<vector<int>> order) {
    bool answer = true;
    vector<int> count(n, 1); count[0] = 0; // 자신을 가리키고 있는 노드의 개수를 저장
    vector<vector<int>> graph(n);

    // 그래프 생성
    for(const vector<int>& edge : path){
        graph[edge[0]].push_back(edge[1]);
        graph[edge[1]].push_back(edge[0]);
    }
    // 의존성 저장
    for(const vector<int>& edge : order){
        graph[edge[0]].push_back(edge[1]);
        count[edge[1]]++;
    }
    
    queue<int> q;
    q.push(0);
    
    while(!q.empty()){
        int current = q.front();
        q.pop();
        
        for(const int next : graph[current]){
            if(--count[next] == 0){
                q.push(next);
                n--;
            }
        }
    }
    
    return n==1;
}
```

* **트리**, **위상 정렬?** 문제
* 문제에서 주어지는 그래프는 트리 구조임
* 어떤 노드를 방문해야 다음 노드를 방문할 수 있다는 개념은 보통 **위상 정렬**에서 많이 사용되는 개념
* 해당 문제의 조건은 어떤 노드의 모든 부모를 방문했다면, 그 노드도 방문할 수 있다는 것
* 트리의 노드는 하나의 부모만을 가지므로, `count`를 1로 초기화
* 이후 `order`의 의존성을 단방향 간선으로 추가
* BFS를 돌면서 방문한 노드의 자식들의 `count`를 빼주고 `count`가 `0`이 되면 방문 가능한 상태로 바뀐 것이므로 `queue`에 추가
* BFS가 종료되었을 때, 방문하지 못한 노드가 있으면 모든 방을 탐험 못 하는 것
* BFS가 진행되다 보면 `count`가 **음수**가 되는 경우가 분명히 존재하지만, 이는 **이미 방문 처리된 노드**이기에 **결과에 영향을 주지 않음**

### [가사 검색](https://school.programmers.co.kr/learn/courses/30/lessons/60060)

```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;

// Trie 노드
class Node {
public:
    Node* child[26];
    int count;
    Node(){
        count = 0;
        for(int i=0; i<26; i++) child[i] = nullptr;
    }
    
    ~Node(){
        for(int i=0; i<26; i++) 
            delete child[i];
    }
    
    // 삽입 연산
    void insert(const string& str){
        Node* curr = this;                  
        for(int i=0; i<str.length(); i++){
            curr->count++;                  // 현재 노드의 count를 증가
            int next = str[i] - 'a';
            if(curr->child[next] == nullptr) 
                curr->child[next] = new Node(); 
            curr = curr->child[next];       
        }
    }
    
    
    // count를 반환
    int getCount(const string& str){
        Node* curr = this;
        for(int i=0; i<str.length(); i++){
            if(str[i] == '?') return curr->count;       // ?를 만나면 현재 노드의 count를 리턴
            int next = str[i] - 'a';            
            if(curr->child[next] == nullptr) return 0;  // 다음 노드가 없으면 해당 쿼리가 words에 없는거임
            curr = curr->child[next];
        }
    }
};



vector<int> solution(vector<string> words, vector<string> queries)
{
    vector<int> answer;
    vector<Node*> Trie(10001), TrieReverse(10001);  // 접미사를 찾는 경우 반대로 찾아야 함
    
    // Trie 생성
    for(string& word : words){
        int length = word.length();
        
        // 각 루트는 해당 길이의 문자열이 몇개 있는가를 나타냄
        if(Trie[length] == nullptr) Trie[length] = new Node();
        if(TrieReverse[length] == nullptr) TrieReverse[length] = new Node();
        
        Trie[length]->insert(word);
        reverse(word.begin(), word.end());
        TrieReverse[length]->insert( word);
    }
    
    // queries 탐색
    for(string& querie : queries){
        int length = querie.length();
        
        if(querie[0] != '?'){
            if(Trie[length] == nullptr) answer.push_back(0);
            else answer.push_back(Trie[length]->getCount(querie));
        }
        
        else{
            reverse(querie.begin(), querie.end());
            if(TrieReverse[length] == nullptr) answer.push_back(0);
            else answer.push_back(TrieReverse[length]->getCount(querie));
        }
    }
    
    for(int i=0; i<10001; i++){
        delete Trie[i];
        delete TrieReverse[i];
    }
    return answer;
}
```

* **Trie** 자료구조를 활용하여 해결
* 문제를 보자마자 **Trie**를 쓰는 문제구나 했지만, 다른 방법은 안 될까 고민을 해봄
* 문제의 조건 중 `전체 가사 단어 길이의 합은 2 이상 1,000,000 이하입니다.` 라는 조건 때문에 모두 확인하는 것도 가능할지도? 라는 생각에 `해시맵`을 이용해서 풀어봄
* 시간을 최대한 줄이기 위해 여러 방법을 시도해 봤지만, 전부 효율성 테스트 4, 5번에서 시간 초과
* 아마 단어의 개수는 적지만, 문자열 길이가 매우 긴 경우에 시간 초과가 나는 듯
* 아래는 **해시맵**을 활용해서 풀어본 코드임

```cpp
#include <string>
#include <vector>
#include <unordered_map>
using namespace std;

vector<int> solution(vector<string> words, vector<string> queries) {
    vector<int> answer(queries.size());
    unordered_map<string, int> count;
    string temp1, temp2, q(10000, '?');
    temp1.reserve(10000); temp2.reserve(10000);
    
    for(const string& word : words){
        for(int i=0; i<word.length(); i++){
            temp1.clear(); temp2.clear();
            temp1.append(word, 0, i); temp1.append(q, 0, word.length()-i);
            temp2.append(q, 0, i); temp2.append(word, i, word.length()-i);
            count[temp1]++; count[temp2]++;
        }
    }

    for(int i=0; i<queries.size(); i++) answer[i] = count[queries[i]];
    return answer;
}
```

## 3. C++ Class 와 Struct의 차이

* 모의 면접 질문으로 **Class 와 Struct의 차이점**에 대해 설명하는 것을 과제로 받아 조사해봄

* 조사하기 전 **사전 지식**으로는 `Struct`는 `C`에서 가져온 것이다, `C`의 `Struct`를 **확장**한게 `C++`의 `Class`이다, 기본 형식 지정자가 `Struct는 Public`,` Class는 Private` 이다, `Struct`는 단순히 **데이터를 묶는 용도**이고, `Class`는 **상속 및 다형성**, **함수 선언**, **연산자 오버로딩** 등의 기능을 제공한다 정도로 생각하고 있었음

* 하지만, 놀랍게도 `C++`에서 `Class`와 `Struct`는 기본 형식 지정자를 제외하고는 지원하는 기능, 어셈블리까지 **모든 것이 동일**했음

* 내가 알고 있었던 차이점은 `C`의 `Struct`와 `C++`의 `Class` 간의 차이점임

### 접근 지정자 예제 코드

```cpp
struct TestStruct{
    int value;  // public
};

class TestClass{
    int value;  // private
};

int main() {
    TestStruct testStruct;
    TestClass testClass;

    testStruct.value = 10;      // 접근 가능
    //testClass.value = 10;     // 접근 불가(컴파일 에러)
    return 0;
}
```
* `Struct`의 경우 기본 접근 지정자가 `public`이기에 `main()`에서 `value`에 접근할 수 있지만, `Class`는 기본 접근 지정자가 `private`이라 접근 불가

* 기본 접근 지정자 외에는 `Class`와 `Sturct`가 모두 동일하지만, **접근 지정자는 명시하는 것을 권장**하기에 사실상 큰 차이는 없음

* `Class`의 기본 접근 지정자가 `private`인 이유는 **OOP**에서 **캡슐화**를 지향하기 때문

### 생성자 및 소멸자, 함수, 상속, 오버 라이딩 예제 코드

```cpp
#include <string>
#include <iostream>

struct BaseStruct {
    int value = 10;

    BaseStruct() {
        std::cout << "BaseStruct 생성자" << std::endl;
    }

    virtual ~BaseStruct() {
        std::cout << "BaseStruct 소멸자" << std::endl;
    }

    virtual void MultiplyValue() {
        value *= 2;
    }
};

struct DerivedStruct : BaseStruct {
private:
    std::string name = "Name";

public:
    DerivedStruct() {
        std::cout << "DerivedStruct 생성자" << std::endl;
    }

    ~DerivedStruct() {
        std::cout << "DerivedStruct 소멸자" << std::endl;
    }

    void MultiplyValue() override {
        value *= 3;
    }
};

int main() {
    BaseStruct baseStruct;
    DerivedStruct derivedStruct;

    std::cout << "\nBaseStruct" << std::endl;
    std::cout << "기본 값: " << baseStruct.value << std::endl;
    baseStruct.MultiplyValue();
    std::cout << "MultiplyValue(x2) 후: " << baseStruct.value << std::endl;

    std::cout << "\nDerivedStruct" << std::endl;
    std::cout << "기본 값: " << derivedStruct.value << std::endl;
    
    derivedStruct.MultiplyValue(); 
    std::cout << "MultiplyValue(x3, 오버라이딩) 후: " << derivedStruct.value << std::endl << std::endl;

    // std::cout << derivedStruct.name << std::endl; // 컴파일 에러
    return 0;
}

// 출력 결과
//
// BaseStruct 생성자
// BaseStruct 생성자
// DerivedStruct 생성자
//
// BaseStruct
// 기본 값: 10
// MultiplyValue(x2) 후: 20

// DerivedStruct
// 기본 값: 10
// MultiplyValue(x3, 오버라이딩) 후: 30

// DerivedStruct 소멸자
// BaseStruct 소멸자
// BaseStruct 소멸자

```
* 모두 `Class`와 동일하게 동작한다.

* `BaseStruct`의 `value`가 접근 지정자가 지정되지 않았으므로, `public`으로 처리되고 이를 상속 받은 `DerivedStruct`에서도 `public`으로 접근할 수 있음

### Struct와 Class 선택
* 사실 기술적인 부분에서 `Sturct`와 `Class`는 동일하다고 봐도 무방함
* 하지만 어떤 것을 선택했느냐에 따라 **의도**를 파악하기 쉽고 **가독성**이 좋아짐
* 일반적으로 데이터만을 저장하는 경우 `Sturct`, 데이터와 로직을 함께 포함하는 경우 `Class`를 선택
