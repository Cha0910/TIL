### [주차 요금 계산](https://school.programmers.co.kr/learn/courses/30/lessons/92341)

```cpp
#include <string>
#include <vector>
#include <sstream>
#include <map>
#include <cmath>

using namespace std;

// 문자열 나누기
vector<string> split(const string& str, char c){
    vector<string> result;
    string temp;
    stringstream ss(str);
    while(getline(ss, temp, c))
        result.push_back(temp);
    return result;
}

// 문자열 시간을 int로
int getTime(const string& str){
    vector<string> data = split(str, ':');
    return 60 * stoi(data[0]) + stoi(data[1]);
}

vector<int> solution(vector<int> fees, vector<string> records) {
    vector<int> answer;
    map<string, int> map, park;
    
    // 입차, 출차 시간을 계산해 각 자동차 별로 map에 주차 시간을 저장
    for(const string& record : records){
        vector<string> data = split(record, ' ');
        int time = getTime(data[0]);
        if(data[2] == "IN") map[data[1]] = time;
        else {
            park[data[1]] += time - map[data[1]];   
            map[data[1]] = -1;
        }
    }
    
    // 주차 시간을 통해 요금 계산
    for(auto [number, time] : map){
        if(time != -1) park[number] += getTime("23:59") - time;
        int price = fees[1];
        if(park[number] > fees[0])
            price += ceil((park[number] - fees[0]) / (double)fees[2]) * fees[3];
        answer.push_back(price);
    }
    
    return answer;
}
```

* `std::map`을 활용하여 푸는 문제
* 자동차 번호를 오름차순으로 출력해야 하기 때문에 `std::unordered_map`이 아닌 `std::map`을 사용해야 함
* 주차 시간이 단위 시간으로 나누어 떨어지지 않으면 올림 해야 하므로 `std::ceil()`을 사용
* `park[]`와 `fees[]`가 둘 다 **int**형 벡터이므로 하나를 **double**로 형변환 해줘야 올림 연산이 정상적으로 됨

### [표 편집](https://school.programmers.co.kr/learn/courses/30/lessons/81303)

```cpp
#include <string>
#include <vector>
#include <stack>

using namespace std;

string solution(int n, int k, vector<string> cmd) {
    string answer(n, 'O');
    int cur = k;
    stack<int> deleted;
    vector<int> prev(n), next(n);
    
    // 이중 연결 리스트
    for(int i=0; i<n; i++){
        prev[i] = i-1;
        next[i] = i+1;
    }
    next[n-1] = -1;
    
    for(const string& command : cmd){ 
        
        // move 만큼 위로 이중 연결 리스트의 앞으로 이동
        if(command[0] == 'U'){
            int move = stoi(command.substr(2));
            while(move > 0){
                cur = prev[cur];
                move--;
            }
        }
        
        //move 만큼 위로 이중 연결 리스트의 뒤로 이동
        else if(command[0] == 'D'){
            int move = stoi(command.substr(2));
            while(move > 0){
                cur = next[cur];
                move--;
            }
        }
        
        // 이중 리스트 삭제 후 연결
        else if(command[0] == 'C'){
            deleted.push(cur);
            if(prev[cur] != -1) next[prev[cur]] = next[cur];
            if(next[cur] != -1) prev[next[cur]] = prev[cur];
            
            if(next[cur] != -1) cur = next[cur];
            else cur = prev[cur];
        }
        
        // 복구가 스택으로 진행되기 때문에, 복구 목표인 행의 위 아래 행은 무조건 유효함
        else if(command[0] == 'Z'){
            int index = deleted.top();
            deleted.pop();
            if(prev[index] != -1) next[prev[index]] = index;
            if(next[index] != -1) prev[next[index]] = index;
        }
    }
    
    // 삭제 스택에 있는 인덱스는 X로 변경
    while(!deleted.empty()){
        answer[deleted.top()] = 'X';
        deleted.pop();
    }
    
    return answer;
}
```

* **이중 연결 리스트**와 **스택**으로 해결
* 삭제 및 복구가 스택으로 진행되기 때문에 복구할 때 양옆의 노드가 무조건 유효함
* 처음에는 커서 이동을 모두 합한 값이 `1,000,000 이하`라는 조건이 있어서 벡터로 구현했었지만, 효율성 테스트에서 실패해 이중 연결 리스트로 수정함
* 아마 삭제된 행이 연속적으로 있는 경우에서 실패하는 듯

### [자물쇠와 열쇠](https://school.programmers.co.kr/learn/courses/30/lessons/60059)

```cpp
#include <string>
#include <vector>

using namespace std;

// 행렬 90도 회전
vector<vector<int>> rotate(const vector<vector<int>>& key){
    int m = key.size();
    vector<vector<int>> result(m, vector<int>(m));
    for(int i=0; i<m; i++){
        for(int j=0; j<m; j++){
            result[i][j] = key[m-1-j][i];
        }
    }
    return result;
}

// 현재 열쇠로 자물쇠를 열 수 있는지 판단
bool check(const vector<vector<int>>& lock, const vector<vector<int>>& key, int y, int x, int target){
    int count = 0;
    for(int i=0; i<key.size(); i++){
        for(int j=0; j<key.size(); j++){
            if(key[i][j] == 1 && lock[i+y][j+x] == 1) return false;
            if(key[i][j] == 1 && lock[i+y][j+x] == 0) count++;
        }
    }  
    return target == count;
}

bool solution(vector<vector<int>> key, vector<vector<int>> lock) {
    bool answer = true;
    int M = key.size();
    int N = lock.size();
    int count = 0;  // 자물쇠에 몇 개의 홈이 있는지

    // 열쇠와 자물쇠를 비교하기 위해 벡터를 확장
    vector<vector<int>> compare(N + (M-1)*2, vector<int>(N + (M-1)*2, -1));
    
    // 확장한 벡터에 자물쇠를 저장
    for(int i=0; i<N; i++){
        for(int j=0; j<N; j++){
            compare[i+M-1][j+M-1] = lock[i][j];
            if(lock[i][j] == 0) count++;
        }
    }
    
    // 90도, 180도, 270도 돌린 열쇠
    vector<vector<int>> key1 = rotate(key);
    vector<vector<int>> key2 = rotate(key1);
    vector<vector<int>> key3 = rotate(key2);
    
    for(int i=0; i<N+M-1; i++){
        for(int j=0; j<N+M-1; j++){
            // 0도, 90도, 180도, 270도 돌린 열쇠 중 하나라도 자물쇠를 열 수 있으면
            if(check(compare, key, i ,j, count) || 
               check(compare, key1, i ,j, count) || 
               check(compare, key2, i ,j, count) || 
               check(compare, key3, i ,j, count)) return true;
        }
    }
    
    // 못열면 false
    return false;
}
```

* 어제 해결한 [고고학 최고의 발견](https://school.programmers.co.kr/learn/courses/30/lessons/131702)과 유사한 문제
* 열쇠의 **회전을 고정**시킨 후 회전한 열쇠로 자물쇠를 열 수 있는지 **완전탐색**
* 열쇠를 이동하며 탐색해야 하므로 자물쇠를 열쇠의 크기에 맞춰 확장해야 함

### [여행경로](https://school.programmers.co.kr/learn/courses/30/lessons/43164)

```cpp
#include <string>
#include <vector>
#include <unordered_map>
#include <algorithm>

using namespace std;
void DFS(unordered_map<string, vector<string>>& graph, const string& departure, 
         vector<string>& route, bool& flag, const int& numTickets){
    if(!flag) return;
    
    // 가장 먼저 발견한 모든 티켓을 사용한 루트가 조건에 만족하는 루트임
    if(route.size() == numTickets+1){
        flag = false;
        return;
    }
    
    // 현재 연결된 노드 중 사전 순으로 선택해서 DFS
    for(int i=0; i<graph[departure].size(); i++){
        string temp = graph[departure][i];
        route.push_back(temp);
        
        // 티켓을 사용한 후 DFS
        graph[departure].erase(graph[departure].begin()+i);
        DFS(graph, temp, route, flag, numTickets);
        
        if(!flag) return; // 정답을 찾았으면 복구하지 않고 종료
        
        // 그래프 복구
        graph[departure].insert(graph[departure].begin()+i, temp);
        route.pop_back();
    }
}

vector<string> solution(vector<vector<string>> tickets) {
    vector<string> answer = {"ICN"};
    unordered_map<string, vector<string>> graph;
    int numTickets = tickets.size();
    bool flag = true;
    
    // 그래프 생성
    for(const vector<string>& ticket : tickets)
        graph[ticket[0]].push_back(ticket[1]);
    
    // 그래프를 사전순으로 정렬
    for(auto& [departure, arrival] : graph)
        sort(arrival.begin(), arrival.end());
     
    DFS(graph, "ICN", answer, flag, numTickets);
    return answer;
}
```

* **백트래킹** 문제
* 단순히 사전 순으로 빠른 공항만 선택하면 모든 티켓을 사용하지 못하고 갇힐 수 있음
* 갇힌 경우에는 백트래킹 해서 다른 경우를 선택해야 함

### [단어 퍼즐](https://school.programmers.co.kr/learn/courses/30/lessons/1882)

```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int solution(vector<string> strs, string t)
{
    int answer = 0;
    vector<int> dp(t.length()+1, 20001);    // i번 째 위치까지의 문자열을 최소 몇 개의 단어로 만들 수 있는지 저장
    dp[0] = 0;

    for(int i=1; i<dp.size(); i++){
        for(const string& str : strs){
            int target = i - str.length();
            if(target < 0 || str.back() != t[i-1]) continue;
            
            if(t.substr(target, str.length()) == str){
                dp[i] = min(dp[i], dp[target] + 1);
            }
        }
    }
    
    answer = dp[t.length()] == 20001 ? -1 : dp[t.length()];
    return answer;
}
```

* `DP` 문제
* `dp[i]` 에는 문자열 i번째 위치까지 만드는 데 필요한 최소의 단어를 저장함
* **t**의 길이가 최대 `20,000`이고, **strs**의 길이가 최대 `100`이라 제한 시간을 짧게 준 듯
* `str.back() != t[i-1]` 조건을 추가해서 필요 없는 탐색은 하지 않게 함
* `substr()`은 복사 비용이 발생하기 때문에 이를 `compare()`로 수정해도 될 듯

## 3. C++ 배치고사
* 어제 본 C++ 배치고사의 결과에서 감점이 있었음
* 아마 11번 문제에서 감점 된 것 같아 코드를 검토하였고, 두 가지 문제점을 발견

### 버퍼 오버런
```cpp
void Add(const T& InData)
{
    if (Size == Capacity) {
        capacity = Capacity * 2;
        T* NewData = new T[Capacity]; 
        for (int i = 0; i < Size; ++i) {
            NewData[i] = Data[i]; 
        }
        delete[] Data;
        Data = NewData;
    }

    Data[Size++] = InData; // C6386, 버퍼 오버런 발생
}
```

* 위 함수에서 `C6386`, 버퍼 오버런 경고가 발생함
* 로직 상으로는 발생할 수가 없어서 제출 당시에는 무시하고 진행함
* 프로그램이 강제 종료 되지는 않았으니, 실제로 발생한게 아니라 컴파일러가 경고하는 것
* `if(Size < Capacity) Data[Size++] = InData;`와 같이 수정하니 경고가 사라짐

### 논리 연산자 실수, 미정의 동작

```cpp
T& operator[](int Index)
{
    if(Index >= 0 || Index < Size) {
        return Data[Index];
    }
}
``` 

* 아마 감점은 여기서 된듯
* 논리 연산자를 잘못 사용해서 `index` 검사가 제대로 이루어 지지 않음
* `if`문이 실행되지 않았을 때, `return` 값이 정의 되어 있지 않음
* 아래와 같이 수정

```cpp
T& operator[](int Index)
{
    if (Index >= 0 && Index < Size){
        return Data[Index];
    }
    throw std::out_of_range("Index out of bounds");
}
```
