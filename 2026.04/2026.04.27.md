### [멀리 뛰기](https://school.programmers.co.kr/learn/courses/30/lessons/12914)

```cpp
#include <string>
#include <vector>

using namespace std;

long long solution(int n) {
    long long answer = 0;
    vector<long long> dp(n+1 , 0);
    dp[1] = 1; dp[2] = 2;
    for(int i=3; i<dp.size(); i++)
        dp[i] = (dp[i-1] + dp[i-2]) % 1234567;
    
    answer = dp[n];
    return answer;
}
```

* **DP** 문제
* `i 번째` 칸에 오는 방법은 `i-1 번째` 칸에서 `1`칸 이동하거나,` i-2 번째` 칸에서 `2`칸 이동하는 방법의 합
* **피보나치 수열**이랑 수식이 같음

### [튜브의 소개팅](https://school.programmers.co.kr/learn/courses/30/lessons/1839)

```cpp
#include <vector>
#include <queue>
#include <climits>

using namespace std;

vector<vector<vector<long long>>> dijkstra(
    const vector<vector<int>>& time_map, int m, int n, int s){
    
    // [m][n][distance] = 최소 비용
    vector<vector<vector<long long>>> result(
        m, vector<vector<long long>>(
            n, vector<long long>(
                m*n, INT_MAX)));
    
    priority_queue<vector<long long>, vector<vector<long long>>, greater<vector<long long>>> pq;
    
    pq.push({0, 0, 0, 0}); // 이동 거리, 비용, 현재 좌표
    int dx[4] = {0, 0, -1, 1};
    int dy[4] = {-1, 1, 0, 0};
    
    while(!pq.empty()){
        vector<long long> data = pq.top();
        pq.pop();
        long long distance = data[0], weight = data[1];
        int currentY = data[2], currentX = data[3];
        long long new_distance = distance + 1;
        if(new_distance >= m*n) continue;
        
        for(int i=0; i<4; i++){
            int y = currentY + dy[i];
            int x = currentX + dx[i];
            if(x<0 || x>=n || y<0 || y>=m || time_map[y][x] == -1) 
                continue;
            long long new_weight = weight + time_map[y][x];
            
            if(new_weight < result[y][x][new_distance] && new_weight <= s){
                pq.push({new_distance, new_weight, y, x});
                result[y][x][new_distance] = new_weight;
            }
        }
    }
    return result;
}

vector<int> solution(int m, int n, int s, vector<vector<int>> time_map) {
    vector<int> answer(2);
    vector<vector<vector<long long>>> result = dijkstra(time_map, m, n, s);  
    
    for(int i=0; i<m*n; i++){
        if(result[m-1][n-1][i] <= s){
            answer[0] = i;
            answer[1] = result[m-1][n-1][i];
            break;
        }
    }

    return answer;
}
```

* **다익스트라**를 활용해 풀이
* 문제에서 어려운 점은, 단순히 비용이 아니라 **거리를 우선** 시 해아한다는 것
* 따라서 평범한 다익스트라로는 **중간에 경로가 끊길 수 있음**
* 다익스트라 결과를 **3차원 배열로 생성**헤 거리에 따른 비용을 전부 저장
* `[m-1][n-1]` 위치의 값 중 만족하는 값을 반환

### [매칭 점수](https://school.programmers.co.kr/learn/courses/30/lessons/42893)

```cpp
#include <string>
#include <vector>
#include <sstream>
#include <unordered_map>

using namespace std;

vector<string> split(const string& str){
    vector<string> result;
    string temp;
    for(int i=0; i<str.length(); i++){
        if(str[i] >= 'a' && str[i] <= 'z'){
            temp += str[i];
        }
        else if(str[i] >= 'A' && str[i] <= 'Z'){
            temp += str[i] + 'a' - 'A';
        }
        else{
            if(!temp.empty()){
                result.push_back(temp);
                temp = "";
            }
        }
    }
    return result;
}

int solution(string word, vector<string> pages) {
    int answer = 0;
    double max = -1;
    unordered_map<string, double> score;
    vector<string> urls(pages.size());
    
    for(int i=0; i<word.length(); i++){
        if(word[i] < 'a' || word[i] > 'z')
            word[i] += 'a' - 'A';
    }
    
    for(int i=0; i<pages.size(); i++){

        // url 찾기
        int start = pages[i].find("<meta property=\"og:url\" content=\"") + 33;
        int end = pages[i].find("\"", start);
        string url = pages[i].substr(start, end - start);    
        urls[i] = url;

        // 외부 링크 찾기
        vector<string> links;
        start = pages[i].find("<a href=\"");
        while(start != string::npos){
            start += 9;
            end = pages[i].find("\"", start);
            string link = pages[i].substr(start, end - start);
            links.push_back(link);
            start = pages[i].find("<a href=\"", start);
        }
        
        // 기본 점수 계산
        double defaultScore = 0;
        vector<string> words = split(pages[i]);
        for(int j=0; j<words.size(); j++){
            if(words[j] == word)
                defaultScore++;
        }
        score[url] += defaultScore;
        
        // 링크 점수 계산
        for(int j=0; j<links.size(); j++)
            score[links[j]] += defaultScore / links.size();     
    }
    
    for(int i=0; i<urls.size(); i++){
        if(score[urls[i]] > max){
            max = score[urls[i]];
            answer = i;
        }
    }
    
    return answer;
}
```

* **해시맵**을 활용해 풀이
* 사실 문제 자체는 매우 간단하지만, **문자열 파싱**이 복잡함

## 3. 스마트 포인터 추가

### shared_from_this의 필요성
* `shared_ptr`로 생성된 객체 내부에서 자기 자신을 가리키는 `shared_ptr`을 만드는 경우 사용

* 객체 내부에서 `this`를 통해 새로운 `shared_ptr`을 생성하면 별도의 **Control Block**을 갖는 `shared_ptr`가 생성됨

* `this`는 스마트 포인터가 아닌 **C++** 에서 사용되는 **원시 포인터**이기 때문

* 별도의 **Control Block**을 갖기에 새로 생성한 `shared_ptr`은 원본보다 **낮은 레퍼런스 카운트**를 갖게 됨

* 새로 생성한 `shared_ptr`이 해제되면, 원본은 이미 해제된 메모리를 가리키는 **댕글링 포인터**가 발생

* 만약 사용하지 않았다 하더라도 이후에 원본이 해제될 때, **이중 해제** 문제가 발생함

* **예제 코드**
    ```cpp
    #include <iostream>
    #include <memory>

    class Node{
    public:
        void CreateShared(){
            std::shared_ptr<Node> self(this); // this로 생성할 시 별도의 shared_ptr이 생성됨
            
            // 실제로 가리키는 포인터는 두 개지만 새로 생성된 포인터는 하나로 인식
            // Output: Reference count: 1
            std::cout << "Reference count: " << self.use_count() << std::endl;  

        } // 블럭이 끝나면서 메모리가 해제 됨
    };

    int main() {
        std::shared_ptr<Node> ptr1 = std::make_shared<Node>();
        ptr1->CreateShared();
        
        return 0;
    } // 이미 해제된 ptr1에 대해서 다시 해제를 시도해 이중 해제가 발생
    ```

    ```cpp
    #include <iostream>
    #include <memory>

    //  shared_from_this()를 사용하기 위해선 enable_shared_from_this를 상속받아야 함
    class Node : public std::enable_shared_from_this<Node>{
    public:
        void CreateShared(){
            std::shared_ptr<Node> self = shared_from_this(); // shared_from_this()로 안전하게 생성
            
            // 동일한 share_ptr을 공유함
            // Output: Reference count: 2
            std::cout << "Reference count: " << self.use_count() << std::endl;  

        } // 블럭이 끝나도 Reference count가 1이므로 해제하지 않음
    };

    int main() {
        std::shared_ptr<Node> ptr1 = std::make_shared<Node>();
        ptr1->CreateShared();
        
        return 0;
    } // Reference count가 0이 되어 자동으로 해제
    ```

### 스마트 포인터의 크기
* 기존의 **원시 포인터**는 32비트 환경에서 `4바이트`, 64비트 환경에서 `8바이트`의 크기를 가짐

* `unique_ptr`은 **원본 객체의 주소**만을 갖고 있으면 충분하기에, 원시 포인터와 **동일한 크기**를 가짐

* `share_ptr`과 `weak_ptr`은 **원본 객체의 주소**와 **Control Block**의 주소를 함께 갖고 있어야 하기에, **두 개의 포인터**를 가지며 크기 또한 **두 배의 크기**를 가짐

## 4. 객체의 복사

### 객체의 복사를 막는 이유
* **객체**는 큰 크기를 가질 수 있으며, 이를 모두 복사하는 것은 **매우 큰 비용**을 필요로 함

* 객체 내부에 **포인터**가 존재할 경우, 포인터가 가리키는 객체가 복사되는 것이 아닌 **포인터만 복사**하기에 하나의 객체가 소멸하면 남은 객체는 **댕글링 포인터** 문제가 발생함

* **파일 스트림**, **mutex**, **싱글톤**과 같이 유일하게 존재해야 하는 객체들이 있음

### 객체의 복사를 막는 방법

1. 복사 생성자와 대입 연산자를 **private**으로 선언

    ```cpp
    #include <iostream>
    #include <string>

    class People {
    private:
        std::string name;

        // 복사 생성자와 대입 연산자를 private에 선언
        People(const People&);
        People& operator=(const People&);

    public:
        People(std::string _name) : name(_name) {
            std::cout << name << " 생성" << std::endl;
        }

        ~People(){
            std::cout << name << " 소멸" << std::endl;
        }
    };

    int main() {
        People people1("Kim");

        // 접근 제한자 범위로 인해 컴파일 에러
        //People people2(people1);
        //People people3 = people1;
            
        return 0;
    } 
    ```
    * 복사 생성자와 대입 연산자를 `private`으로 선언하여 외부에서의 접근을 차단
    * 하지만 이는 내부에서는 호출할 수 있으며, `friend` 키워드를 사용하면 외부에서도 호출할 수 있음
    * 만약 **구현부를 작성**했다면, 위 방법으로 복사가 가능해짐

2. 복사 생성자와 대입 연산자 **명시적으로 비활성화**

    ```cpp
    #include <iostream>
    #include <string>

    class People {
    private:
        std::string name;

    public:
        // 복사 생성자와 대입 연산자를 delete 처리
        People(const People&) = delete;
        People& operator=(const People&) = delete;

        People(std::string _name) : name(_name) {
            std::cout << name << " 생성" << std::endl;
        }

        ~People(){
            std::cout << name << " 소멸" << std::endl;
        }
    };

    int main() {
        People people1("Kim");

        // People people2(people1);
        // People people3 = people1;
            
        return 0;
    } 
    ```
    * `C++11` 이상에서 지원하는 방법
    * `private` 선언 방식의 단점을 개선하여 명시적으로 비활성화하기에 **어떤 방법으로도 호출할 수 없음**
    * 복사 생성자와 대입 연산자를 `public`으로 선언해 컴파일 시 복사 및 대입이 불가능함을 **에러 메시지**로 보여줌

3. **이동**만 허용
    ```cpp
    #include <iostream>
    #include <string>
    #include <utility>

    class People {
    private:
        std::string name;

    public:
        // 복사 생성자와 대입 연산자를 delete 처리
        People(const People&) = delete;
        People& operator=(const People&) = delete;

        // 이동 생성자 및 이동 대입 연산자를 default 처리
        People(People&& other) noexcept = default;
        People& operator=(People&& other) noexcept = default;

        People(std::string _name) : name(_name) {
            std::cout << name << " 생성" << std::endl;
        }

        ~People(){
            std::cout << name << " 소멸" << std::endl;
        }
    };

    int main() {
        People people1("Kim");  // Output: Kim 생성

        // std::move()를 통해 people1, people2를 rvalue로 바꿔줌
        People people2(std::move(people1));
        People people3 = std::move(people2);
           
        // std::move() 함수를 사용하지 않으면 
        // delete한 복사 생성자, 대입 연산자가 실행되므로 컴파일 에러
        // People people4(people3);
        // People people5 = people4;

        return 0;
    } 
    // people1 -> people2 -> people3 로 소유권이 이전 되었으므로 
    // people1, people2는 빈 문자열을 가짐
    // Output
    // Kim 소멸
    //  소멸
    //  소멸
    ```
    * 이동 생성자와 이동 대입 연산자를 통해 **소유권**을 옮김
    * **스마트 포인터**의 `unique_ptr`와 사용 방법이 동일한데, 같은 방식으로 설계된 것
    * `std::move()` 함수를 명시적으로 작성해야 동작하기에, 안정성과 가독성이 높음
