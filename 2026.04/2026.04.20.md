### [괄호 회전하기](https://school.programmers.co.kr/learn/courses/30/lessons/76502)

```cpp
#include <string>
#include <vector>
#include <stack>
using namespace std;

// 입력받은 문자열이 올바른 괄호 문자열인지
bool check(const string& str){
    stack<char> s;
    
    for(int i=0; i<str.length(); i++){

        // 닫는 괄호이고
        if(str[i] == ']' || str[i] == ')' || str[i] == '}'){
            // 짝이 맞으면
            if(!s.empty() && abs(s.top() - str[i]) <=2) {
                s.pop();    // 스택에서 뺌
                continue;
            }
            else return false;  // 짝이 틀리면 잘못된 문자열
        }
        else s.push(str[i]);    // 여는 괄호면 무조건 스택에 넣음
    }
    
    return s.empty();   // 마지막에 스택이 비어있으면 올바른 문자열
}

int solution(string s) {
    int answer = 0;
    answer += check(s);

    // 가장 앞에 있는 것을 뒤에 붙임
    for(int i=0; i<s.length()-1; i++){
        s.push_back(s[0]);
        s.erase(0, 1);
        answer += check(s);
    }
    return answer;
}
```

* **스택**을 활용하는 문제
* 코테 단골 유형인 **괄호 쌍**을 맞추는 문제
* 스택을 활용하여 괄호가 올바른지 구분 하면 됨
* 문자열을 회전할 때, `erase()` 함수를 사용하기 때문에 `O(n^2)`의 시간 복잡도를 가지지만, 문자열의 최대 길이가 `1,000`이므로 충분함
* **deque**를 사용하면 `O(n)` 시간으로 회전 가능

### [몸짱 트레이너 라이언의 고민](https://school.programmers.co.kr/learn/courses/30/lessons/1838)

```cpp
#include <vector>
#include <cmath>
using namespace std;

// 해당 락커에 배치가 가능한지 여부
bool isCan(vector<pair<int, int>>& locker, int& minDist, int& i, int& j){
    for(const pair<int, int>& pos : locker){
        int dist = abs(pos.first - i) + abs(pos.second - j);
        if(dist < minDist) return false;
    }
    return true;
}

// size의 locker에 num명을 minDist만큼의 거리로 배치가 가능한지 여부
bool check(int size, int& num, int& minDist){
    for(int i=0; i<size; i++){
        for(int j=0; j<size; j++){
            vector<pair<int, int>> locker;
            int count = 1;
            locker.push_back({i, j});
            
            for(int k=i; k<size; k++){
                for(int l=0; l<size; l++){
                    if(isCan(locker, minDist, k, l)){
                        locker.push_back({k, l});
                        count++;
                        if(count == num) return true;
                    }
                }
            }
        }
    }
    return false;
}


// 전역 변수를 정의할 경우 함수 내에 초기화 코드를 꼭 작성해주세요.
int solution(int n, int m, vector<vector<int>> timetable) {
    int answer = 0;
    
    // 최대 몇 명 겹치는지 확인
    int maxCount = 1;
    for(int t=600; t<=1320; t++) { // 운영 시간 범위
        int count = 0;
        for(const vector<int>& time : timetable)
            if (t >= time[0] && t <= time[1]) count++;
        maxCount = max(maxCount, count);
    }
    if(maxCount == 1) return 0;
    
    // 이분 탐색으로 최댓값 찾기
    int right = (n-1)*2, left = 1;
    while(left <= right){
        int mid = (left + right) / 2;
        if(check(n, maxCount, mid)){
            answer = mid;
            left = mid + 1;
        }
        else right = mid - 1;
    }
    return answer;
}
```

* **그리디**, **이분 탐색**을 활용해 풀이
* 시간의 범위가 `600 ~ 1320` 밖에 안되므로, 각 시간에 몇 명이 겹치는지 직접 세도 충분함
* 만약 시간 범위가 더 크다면, **누적합**을 활용하면 계산 가능
* 락커의 사이즈와 겹치는 손님의 수는 고정되어 있으므로, **가장 가까운 손님의 거리 또한 고정** 됨
* **이분 탐색**을 활용하여 **가장 가까운 손님의 거리**를 찾음
* `n`의 크기가 `최대 10` 밖에 안되므로 `O(n^4)`로 모든 경우를 찾아도 시간은 충분함

## 3. malloc() 관련 추가 학습

### void
* 데이터 타입이 정의되지 않았음을 알리는 **키워드**
* `void 변수`는 생성 불가능하지만, `void 포인터`는 생성 가능(`malloc()`의 반환 타입)
* `void 포인터`는 어떤 타입의 포인터든 담을 수 있지만, **역참조** 시 컴파일 에러

### malloc() 초기화 방법
* `malloc()`은 메모리만 할당해주고 초기화는 실행하지 않기에 쓰레기 값이 저장되어 있음
*  이를 초기화 하는 방법은 아래의 3가지 방법이 있음

1. **calloc()**

    ```cpp
    #include <iostream>
    using namespace std;
    
    int main() {
        int *ptr = (int*)calloc(5, sizeof(int)); // 5개 공간을 할당과 동시에 0으로 초기화

        for(int i=0; i<5; i++)
            std::cout << *(ptr+i) << " "; // Output: 0 0 0 0 0;

        free(ptr);
        ptr = nullptr;

        return 0;
    }
    ```

2. **memset()**

    ```cpp
    #include <iostream>
    #include <cstring>
    using namespace std;
        
    int main() {
        char *ptr = (char*)malloc(5 * sizeof(char)); // 5개 공간을 할당
        memset(ptr, 'c', 5 * sizeof(char));          // 5개 공간을 c로 초기화
            
        for(int i=0; i<5; i++)
            std::cout << *(ptr+i) << " "; // Output: c c c c c;

        free(ptr);
            
        ptr = nullptr;
        return 0;
    }
    ```

3. **반복문**

    ```cpp
    #include <iostream>
    using namespace std;
    
    int main() {
        int *ptr = (int*)malloc(5 * sizeof(int)); // 5개 공간을 할당

        for(int i=0; i<5; i++)
            *(ptr+i) = i;

        for(int i=0; i<5; i++)
            std::cout << *(ptr+i) << " "; // Output: 0 1 2 3 4
        free(ptr);
        ptr = nullptr;

        return 0;
    }
    ```
