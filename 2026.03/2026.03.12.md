### [호텔 대실](https://school.programmers.co.kr/learn/courses/30/lessons/155651)
```cpp
#include <string>
#include <vector>
#include <sstream>
#include <algorithm>

using namespace std;

// 문자열 나누기
vector<string> split(const string& str, char c){
    vector<string> result;
    stringstream ss(str);
    string temp;
    while(getline(ss, temp, c)){
        result.push_back(temp);
    }
    return result;
}

// HH:MM 형태의 시간을 분 단위로 변환
int timeToInt(const vector<string>& str){
    int result = stoi(str[0]) * 60 + stoi(str[1]);
    return result;
}
 
int solution(vector<vector<string>> book_time) {
    int answer = 0;
    vector<pair<int, int>> book_time_int;   // 각 예약의 시작 시간, 종료 시간 저장
        
    for(vector<string>& book : book_time){
        vector<string> result = split(book[0], ':');
        int startTime = timeToInt(split(book[0], ':'));
        int endTime = timeToInt(split(book[1], ':')) + 9;   // 청소시간 더해줌
        book_time_int.push_back({startTime, endTime});
    }
    
    sort(book_time_int.begin(), book_time_int.end());
    
    for(const pair<int, int>& book : book_time_int){
        int count = 0;
        
        for(const pair<int, int>& other_book : book_time_int){
            // 대실 종료 시각에 다른 방의 대실 시간과 겹치면
            if(other_book.first <= book.second && book.second <= other_book.second )
                count++;    // 필요한 방의 개수를 늘림
        }

        answer = count > answer ? count : answer; // 최대값이면 저장
    }
    return answer;
}
```

### [지형 이동](https://school.programmers.co.kr/learn/courses/30/lessons/62050)
```cpp
#include <string>
#include <vector>
#include <queue>
#include <cmath>
#include <map>

using namespace std;

int solution(vector<vector<int>> land, int height) {
    int answer = 0;
    int num_land = 0;   // 서로 다른 색상의 지형 개수
    
    vector<vector<int>> node(land.size(), vector<int>(land.size(), -1));
    
    queue<pair<int,int>> q;
    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};
    
    // BFS를 통해 사다리 없이 이동 가능한 칸을 동일한 번호의 노드로 나누기
    for(int i=0; i<land.size(); i++){
        for(int j=0; j<land.size(); j++){
            if(node[i][j] == -1){
                q.push({i, j});
                num_land++;
                node[i][j] = num_land;
            }
                
            while(!q.empty()){
                pair<int, int> pos = q.front();
                q.pop();
                
                for(int k=0; k<4; k++){
                    int x = pos.first + dx[k];
                    int y = pos.second + dy[k];
                    
                    if(x>=0 && x<land.size() && y>=0 && y<land.size() && 
                       node[x][y] == -1 && abs(land[pos.first][pos.second] - land[x][y]) <= height){
                        q.push({x, y});
                        node[x][y] = num_land;
                    }
                }
            }
        }
    }
   
    // 나누어진 노드로 그래프 생성 graph[현재 노드][다음 노드] = 가중치
    vector<map<int, int>> graph(num_land); 
    for(int i=0; i<node.size(); i++){
        for(int j=0; j<node.size(); j++){
            int currentNode = node[i][j]-1;
            for(int k=0; k<4; k++){
                int x = i + dx[k];
                int y = j + dy[k];
                
                if(x>=0 && x<node.size() && y>=0 && y<node.size()){
                    int otherNode = node[x][y]-1;
                    
                    if(currentNode != otherNode){
                        int weight = abs(land[i][j] - land[x][y]);
                    
                        if(graph[currentNode].find(otherNode) == graph[currentNode].end() 
                           || graph[currentNode][otherNode] > weight)
                            graph[currentNode][otherNode] = weight;
                    }
                }
            }    
        }
    }
      
    // 프림 알고리즘으로 최소 신장트리
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    vector<bool> visit(graph.size(), false);
    pq.push({0, 0});
    
    while(!pq.empty()){
        int currentNode = pq.top().second;
        int weight = pq.top().first;
        pq.pop();
        if(visit[currentNode]) continue;
        visit[currentNode] = true;
        answer += weight;

        for (const auto& edge : graph[currentNode]) {
            if (!visit[edge.first]) {
                pq.push({edge.second, edge.first});
            }
        }
    }

    return answer;
}
```

* 사다리 없이 이동이 가능한 칸은 비용이 0 이므로 동일한 칸이라고 봐도 됨
* BFS로 사다리 없이 이동 가능한 칸을 동일한 노드로 묶음
* 그후 모든 칸을 돌면서 간선 데이터를 **최소값**만 저장하기 위해 **map**에 저장
* 생성된 그래프를 **프림 알고리즘**을 통해 **최소 신장 트리**의 비용을 계산

### [오픈채팅방](https://school.programmers.co.kr/learn/courses/30/lessons/42888)
```cpp
#include <string>
#include <vector>
#include <map>
#include <sstream>

using namespace std;

// 문자열 나누기
vector<string> split(const string& str, char c){
    vector<string> result;
    stringstream ss(str);
    string temp;
    while(getline(ss, temp, c))
        result.push_back(temp);
    return result;
}

vector<string> solution(vector<string> record) {
    vector<string> answer;
    map<string, string> uidMap; // uidMap[uid] = 닉네임
    
    // uidMap 생성
    for(const string& rog : record){
        vector<string> data = split(rog, ' ');
        if(data[0] == "Enter" || data[0] =="Change")    // 채팅방을 들어오거나 닉네임을 변경하면
            uidMap[data[1]] = data[2];                  // uid, 닉네임 업데이트
    }
    
    // 업데이트 된 uid 로 record를 출력
    for(const string& rog : record){
        vector<string> data = split(rog, ' ');
        if (data[0] == "Change") continue;
        
        string result = uidMap[data[1]];
        if(data[0] == "Enter") result += "님이 들어왔습니다.";
        else if(data[0] =="Leave") result += "님이 나갔습니다.";
        
        answer.push_back(result);
    }
    
    return answer;
}
```

* `C++ 코딩테스트 완전 정복 : 입문부터 실전 모의고사까지` 수강 중 풀어준 문제
* 해설을 듣기 전 직접 해결한 후 해설과 비교해 봄
* 간단한 문제였기에 금방 해결할 수 있었지만 해설을 보며 문제 해결을 위한 **접근 방식**, **조건 확인**, **테스트 케이스**와 같은 기초적인 부분을 점검할 수 있었음

## 3. C++ 코딩테스트 완전 정복

### stringstream
* **string**을 **입출력 스트림**처럼 사용할 수 있게 하는 클래스
* 다른 데이터 타입을 **string**으로 변환하거나 역으로 **string** 데이터를 다른 타입으로 **변환**할 때 사용
* `split()` 함수와 같은 데이터 파싱을 구현하는 데 사용

### unique()
* **vector**에서 **연속으로 중복되지 않는** 원소들을 앞에서부터 덮어쓰는 함수
* 모든 원소가 유일하게 존재하기 위해서는 **정렬된 상태**의 벡터를 사용해야 함
* 벡터의 크기를 줄이는 것이 아닌 앞에는 중복을 제거한 원소, 뒤에는 쓰레기 값을 갖고 있음
* 쓰레기 값 시작 부분의 **iterator**를 반환하며 주로 `erase()`와 함께 사용

    ```cpp
    #include <iostream>
    #include <vector>
    #include <algorithm> 

    using namespace std;

    int main() {
        vector<int> vec = {1, 5, 2, 2, 3, 4, 1, 5};

        sort(vec.begin(), vec.end());                           // 정렬
        vec.erase(unique(vec.begin(), vec.end()), vec.end());   // 쓰레기 값의 시작점 부터 끝까지 삭제

        for (int n : vec) 
            cout << n << " "; // Output: 1 2 3 4 5
    }
    ```

## 4. 퍼즐 어드벤처 디자인하기

### 승리 조건 확인

* 플레이어 캐릭터가 3개의 열쇠를 모두 보유하고 있는지를 확인한 후 **클리어 여부를 결정하는 로직**을 구현했으며 이를 위해 아래의 세 개의 변수를 추가함
    * **RequiredKeys (Array):** 클리어를 위해 필요한 열쇠를 저장
    * **HeldKeys (Array):** 액터(플레이어)가 보유하고 있는 열쇠를 저장
    * **FoundAllKeys (Boolean):** 최종적으로 모든 열쇠를 가졌는지 판단

* `For Each Loop with Break`, `Contains Item (Array)` 노드로 **RequiredKeys**의 모든 요소가 **HeldKeys**에 포함되어 있는지 확인 

* 하나라도 없으면 **FoundAllKeys**를 False로 설정하고 `For Each Loop with Break` 노드의 **Break**에 연결하여 반복문을 중단

* 반복문이 종료된 후 **FoundAllKeys**가 True면 `Disable Input`으로 플레이어의 입력을 차단한 후 클리어 화면 전환 및 다음 레벨을 로드
