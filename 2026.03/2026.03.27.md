### [2개 이하로 다른 비트](https://school.programmers.co.kr/learn/courses/30/lessons/77885)

```cpp
#include <string>
#include <vector>
#include <bitset>
using namespace std;

vector<long long> solution(vector<long long> numbers) {
    vector<long long> answer;
    for(const long long number : numbers){
        bitset<64> numberBit(number);

        // 짝수인 경우에는 첫번째 비트가 0이므로 1을 더해주면 조건을 만족
        if(number % 2 == 0) {
            answer.push_back(number + 1);
            continue;
        }
        
        // 홀수인 경우에는 가장 처음으로 만난 0을 1로 바꿔준 후
        // 그 바로 이전 비트를 0으로 바꾸는 것이 최솟값
        for(int i=0; i<64; i++){
            if(numberBit[i] == 0) {
                numberBit[i] = 1;
                numberBit[i-1] = 0;
                answer.push_back(numberBit.to_ullong());
                break;
            }
        }   
    }
    return answer;
}
```

* 비트의 표현 방식?을 이해해야 하는 문제
* 짝수인 경우에는 첫 번째 비트가 0이므로 단순히 1만 더하면 원래 수보다 크면서 다른 비트의 수가 1개이므로 가능한 최솟값
* 홀수인 경우에는 처음으로 만난 0을 1로 바꾼 후 바로 이전 위치의 비트를 0으로 바꾸는 것이 최솟값임

### [스타 수열](https://school.programmers.co.kr/learn/courses/30/lessons/70130)

```cpp
#include <string>
#include <vector>
#include <queue>

using namespace std;

int solution(vector<int> a) {
    int answer = -1;
    vector<int> count(500000, 0);   // 숫자의 빈도 수 저장
    if(a.size() == 1) return 0;
    
    for(const int& num : a) count[num]++;
    
    priority_queue<pair<int, int>> pq;  // 가장 높은 빈도의 숫자를 먼저 검사하기 위한 우선순위 큐
    
    for(int i=0; i<count.size(); i++){
        if(count[i] > 0) 
            pq.push({count[i], i});
    }
        
    vector<bool> used(a.size(), false);
    while(!pq.empty()){
        fill(used.begin(), used.end(), false);
        int count = 0;
        int target = pq.top().second;   // 각 집합의 교집합이 될 숫자
        pq.pop();
        
        // 교집합이 될 숫자의 좌우에 집합으로 묶을 수 있는 숫자가 있으면 집합의 개수 증가
        for(int i=0; i<a.size(); i++){
            if(a[i] == target){
                used[i] = true;
                if(i>0 && a[i-1] != target && !used[i-1]){
                    used[i-1] = true;
                    count++;
                }
                else if(i<a.size()-1 && a[i+1] != target && !used[i+1]){
                    used[i+1] = true;
                    count++;
                }
            }
        }
        if(count > answer) answer = count;
        else break;
    }
    return answer*2;
}
```

* 그리디 문제
* 교집합이 될 수를 정한 다음 그 수와 묶을 수 있는 수를 수열에서 빼는 방식
* 최적화를 위해 등장 빈도가 높은 수부터 검사
* 뺀 개수 * 2가 스타 수열의 길이임

### [2차원 동전 뒤집기](https://school.programmers.co.kr/learn/courses/30/lessons/131703)

```cpp
#include <string>
#include <vector>

using namespace std;

void flip(vector<vector<int>>& board, int n){
    if(n < board[0].size()){
        for(int i=0; i<board.size(); i++){
            board[i][n]++;
            board[i][n] %= 2;
        }
    }
    else{
        for(int i=0; i<board[0].size(); i++){
            board[n-board[0].size()][i]++;
            board[n-board[0].size()][i] %= 2;
        }  
    }
}

int solution(vector<vector<int>> beginning, vector<vector<int>> target) {
    int answer = 0, answerReverse = 0;
    vector<vector<int>> board = beginning;
    vector<vector<int>> boardReverse = beginning;
    
    // 1행이 타겟과 동일하게 되게 각 열을 뒤집음
    for(int i=0; i<board[0].size(); i++){
        if(board[0][i] != target[0][i]){
            flip(board, i);
            answer++;
        }
    }
    
    // 1열이 타겟과 동일하게 되게 각 행을 뒤집음
    for(int i=1; i<board.size(); i++){
        if(board[i][0] != target[i][0]){
            flip(board, i + board[0].size());
            answer++;
        }
    }

    // 타겟과 다른 동전이 하나라도 있으면 불가능 한 것
    for(int i=0; i<board.size(); i++){
        for(int j=0; j<board[i].size(); j++){
            if(board[i][j] != target[i][j])
                answer = 21;
        }
    }
    
    // 1행이 타겟과 반대로 되게 각 열을 뒤집음
    for(int i=0; i<boardReverse[0].size(); i++){
        if(boardReverse[0][i] == target[0][i]){
            flip(boardReverse, i);
            answerReverse++;
        }
    }
    
    // 1열이 타겟과 동일하게 되게 각 행을 뒤집음
    for(int i=0; i<boardReverse.size(); i++){
        if(boardReverse[i][0] != target[i][0]){
            flip(boardReverse, i + boardReverse[0].size());
            answerReverse++;
        }
    }

    // 타겟과 다른 동전이 하나라도 있으면 불가능 한 것
    for(int i=0; i<boardReverse.size(); i++){
        for(int j=0; j<boardReverse[i].size(); j++){
            if(boardReverse[i][j] != target[i][j])
                answerReverse = 21;
        }
    }
    
    // 두 경우 중 최솟값이 정답
    answer = min(answer, answerReverse);
    return answer == 21 ? -1 : answer;
}
```

* 동전을 뒤집는 순서는 최종 결과에 아무 영향을 끼치지 않음
* 최종 결과는 어떤 행과 열의 동전들을 뒤집었냐에 따라서 결정됨
* 동일한 행이나 열을 다시 뒤집는 건 초기 상태로 돌아가는 것이므로 아무 의미가 없음
* 따라서 모든 행과 열은 단 한 번만 뒤집을 수 있음
* 즉, 다른 동전들은 신경 쓰지 않고 어떤 행과 열을 기준으로 뒤집으면 target의 상태와 똑같아짐
* 위 코드에선 첫 번째 행과 첫 번째 열을 기준으로 잡아 해결함
* 첫 번째 열을 정답과 반대가 되게끔 뒤집은 후 행을 기준으로 뒤집는 게 더 최소의 경우가 될 수 있으므로 두 가지 경우로 나누어서 최솟값을 구함
