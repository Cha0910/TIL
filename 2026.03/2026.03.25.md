### [가장 큰 수](https://school.programmers.co.kr/learn/courses/30/lessons/42746)

```cpp
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

string solution(vector<int> numbers) {
    string answer = "";
    sort(numbers.begin(), numbers.end(), [](const int A, const int B){
        string a = to_string(A);
        string b = to_string(B);
        return a + b > b + a;
    });
    if(numbers[0] == 0) return "0";
    for(const int& number : numbers)
        answer += to_string(number);
    return answer;
}
```

* **정렬** 문제
* 두 수를 번갈아서 이어 붙인 문자열은 자릿수가 같기에 사전 순 정렬이 더 쉬워짐
* 사전 순으로 더 큰 문자열의 앞 숫자를 앞으로 오게 정렬

### [기둥과 보 설치](https://school.programmers.co.kr/learn/courses/30/lessons/60061)

```cpp
#include <string>
#include <vector>

using namespace std;

// 설치
void Install(vector<vector<int>>& board, const vector<int>& build_frame){
    int x = build_frame[0], y = build_frame[1], type = build_frame[2];
    
    // 기둥
    if(type == 0){
        if(y == 0 || board[y-1][x] & 1 || board[y][x] & 2 || (x>0 && board[y][x-1] & 2)) {
            board[y][x] += 1;
        }
    }
        
    // 보
    else{
        if(board[y-1][x] & 1 || board[y-1][x+1] & 1 ||
           ((x>0 && board[y][x-1] & 2) && board[y][x+1] & 2)){
            board[y][x] += 2;
        } 
    }
}

// x, y 위치의 기둥 또는 보가 유효한가
bool check(const vector<vector<int>>& board, const int& x, const int& y){
    // 현재 칸에 기둥이 있는 경우
    if(board[y][x] & 1){
        if(!(y == 0 || board[y-1][x] & 1 || board[y][x] & 2 || (x>0 && board[y][x-1] & 2))) {
            return false;
        }
    }
    
    // 현재 칸에 보가 있는 경우
    if(board[y][x] & 2){
        if(!((y>0 && board[y-1][x] & 1) || (y>0 && x<board.size()-1 && board[y-1][x+1] & 1) ||
           ((x>0 && board[y][x-1] & 2) && (x<board.size()-1 && board[y][x+1] & 2)))){
            return false;
        } 
    }
    
    return true;
}

// 모든 칸에 대하여 기둥, 보가 유효한지 확인
bool Test(const vector<vector<int>>& board){
    for(int i=0; i<board.size(); i++){
        for(int j=0; j<board[i].size(); j++){
            if(board[i][j] == 0) continue;
            if(!check(board, j, i)) return false;
        }
    }
    return true;
}

// 삭제
void Delete(vector<vector<int>>& board, const vector<int>& build_frame){
    int x = build_frame[0], y = build_frame[1], type = build_frame[2];
    // 기둥
    if(type == 0){
        board[y][x] -= 1;
        if(!Test(board)) board[y][x] += 1;
    } 
        
    // 보
    else{
        board[y][x] -= 2;
        if(!Test(board)) board[y][x] += 2;
    }
}

vector<vector<int>> solution(int n, vector<vector<int>> build_frame) {
    vector<vector<int>> answer;
    vector<vector<int>> board(n+1, vector<int>(n+1, 0)); // 0: 설치안됨, 1: 기둥 설치, 2: 보 설치, 3: 기둥, 보 설치
    
    for(const vector<int>& build : build_frame){
        if(build[3] == 1) Install(board, build);
        else Delete(board, build);
    }
    
    for(int i=0; i<n+1; i++){
        for(int j=0; j<n+1; j++){
            if(board[j][i] == 1 || board[j][i] == 2)
                answer.push_back({i, j, board[j][i]-1});
            else if(board[j][i] == 3){
                answer.push_back({i, j, board[j][i]-3});
                answer.push_back({i, j, board[j][i]-2});
            }
        }
    }
    return answer;
}
```

* 특별한 알고리즘 없이 문제의 조건을 **구현**하는 문제
* 삭제 시 선택된 대상을 먼저 지운 후, 모든 기둥과 보가 **유효한지 검사**하여 삭제 가능 여부를 판단
* 처음에는 삭제한 대상이 영향을 주는 대상만 검사했지만, 코드가 너무 복잡해져 모두 검사하는 것으로 바꿈

### [보행자 천국](https://school.programmers.co.kr/learn/courses/30/lessons/1832)

```cpp
#include <vector>
#include <iostream>

using namespace std;

int MOD = 20170805;

// 전역 변수를 정의할 경우 함수 내에 초기화 코드를 꼭 작성해주세요.
int solution(int m, int n, vector<vector<int>> city_map) {
    int answer = 0;
    vector<vector<int>> dpUp(m, vector<int>(n, 0));
    vector<vector<int>> dpLeft(m, vector<int>(n, 0));
    dpUp[1][0] = 1;
    dpLeft[0][1] = 1;
    
    for(int i=0; i<m; i++){
        for(int j=0; j<n; j++){
            if(city_map[i][j] == 1) continue;
            
            // 위에서 오는 경우
            if(i > 0){
                if(city_map[i-1][j] == 0)
                    dpUp[i][j] += (dpUp[i-1][j] + dpLeft[i-1][j]) % MOD;
                
                else if(city_map[i-1][j] == 2)
                    dpUp[i][j] += dpUp[i-1][j] % MOD;
            }
                
            // 왼쪽에서 오는 경우
            if(j > 0){
                if(city_map[i][j-1] == 0)
                    dpLeft[i][j] += (dpUp[i][j-1] + dpLeft[i][j-1]) % MOD;
                
                else if(city_map[i][j-1] == 2)
                    dpLeft[i][j] += dpLeft[i][j-1] % MOD;
            }
        }
    }
    
    answer = (dpUp[m-1][n-1] + dpLeft[m-1][n-1]) % MOD;
    return answer;
}
```

* 오른쪽 또는 아래로밖에 이동하지 못하므로 **DP**로 해결
* 어떤 칸에 오는 경우는 위에서 오거나 왼쪽에서 오는 경우밖에 없음
* 두 경우를 나누어 다음 DP를 계산할 때 해당 칸의 상태에 따라서 DP 계산을 다르게 함

## 3. std::string

### find()
* 문자열 안에 특정 문자열이나 **문자를 찾는데** 사용
* 입력한 문자 또는 문자열이 존재하면 **첫 번째 인덱스**를 반환
* 찾지 못했을 경우 `std::string::npos`를 반환

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str = "Apple Banana Orange Apple";
    size_t pos = str.find("Banana");
    
    if (pos != string::npos)  cout << "Found at index: " << pos << endl; // Output: 6
    else cout << "Not found" << endl;
      
    // 특정 위치부터 찾기 (index 10 이후부터 Apple 찾기)
    size_t next_pos = str.find("Apple", 10);
    if (next_pos != string::npos) cout << "Found at index: " << next_pos << endl; // Output: 20
     else cout << "Not found" << endl;

    return 0;
}
```

### substr()
* 문자열의 **일부를 추출**하는데 사용
* 문자열의 일부분을 복사하여 **새로운 문자열**을 생성함

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str = "Hello, C++ World!";

    // 인덱스 7부터 3글자 추출
    string sub = str.substr(7, 3); 
    cout << sub << endl; // Output: C++
    
    // 두 번째 인자 생략 시 끝까지 추출
    string end = str.substr(11);
    cout << end << endl; // Output: World!

    return 0;
}
```

### replace()
* 문자열의 일부를 **변환**하는데 사용
* `substr()`과 달리 기존 문자열 자체를 **수정**함

```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string str = "I like Python";

    // Python (인덱스 7, 길이 6)을 C++로 교체
    str.replace(7, 6, "C++");
    
    cout << str << endl;    // Output: I like C++

    return 0;
}
```

### stringstream
* <sstream> 헤더에 정의되어 있으며, 문자열을 마치 **입출력 스트림** 처럼 다룰 수 있게 해줌
* 주로 **split**, **타입 변환**에 사용

```cpp
#include <iostream>
#include <string>
#include <sstream>
using namespace std;

int main() {
    // 공백 기준 분리 (Split)
    string data = "apple 100 cherry";
    stringstream ss(data);
    string word;
    int number;
    
    ss >> word >> number; // 공백을 기준으로 잘라서 변환
    cout << word << " " << number << endl; // Output: apple 100
    
    // int -> string 
    int val = 2026;
    string s_val;
    stringstream ss2;
    ss2 << val; 
    ss2 >> s_val;
    cout << s_val + "0325" << endl; // Output: 20260325

    return 0;
}
```
