### [숫자 변환하기](https://school.programmers.co.kr/learn/courses/30/lessons/154538)

```cpp
#include <string>
#include <vector>
#include <climits>
using namespace std;

int solution(int x, int y, int n) {
    int answer = 0;
    vector<int> dp(y+1, INT_MAX);
    dp[x] = 0;
    
    for(int i=0; i<=y; i++){
        if(dp[i] != INT_MAX){
            if(i+n < dp.size()) dp[i+n] = min(dp[i+n], dp[i]+1);
            if(i*2 < dp.size()) dp[i*2] = min(dp[i*2], dp[i]+1);
            if(i*3 < dp.size()) dp[i*3] = min(dp[i*3], dp[i]+1);
        }
    }
    answer = dp[y];
    if(answer == INT_MAX) answer = -1;
    return answer;
}
```

* `DP`를 활용해 해결
* 어떤 숫자 `a`가 최소 횟수 `b`번으로 만들어진다고 하면, `a+n`, `2*a`, `3*a`는 최소 횟수 `b+1`번으로 만들 수 있음

### [가장 긴 팰린드롬](https://school.programmers.co.kr/learn/courses/30/lessons/12904)

```cpp
#include <iostream>
#include <string>

using namespace std;
int solution(string s)
{
    int answer = 0;
    
    // 문자열 길이가 홀수인 경우
    for(int i=1; i<s.length()-1; i++){
        int mid = i;
        int left = mid, right = mid;
        int count = 0;
        
        if(s[mid-1] == s[mid+1]){
            left--;
            right++;
            count++;
        }
        else continue;
        
        while(left >= 0 && right < s.length()){            
            if(s[left] != s[right]) break;
            count += 2;
            left--;
            right++;
        }
        if(count > answer) answer = count;
    }
    
    // 문자열 길이가 짝수인 경우
    for(int i=1; i<s.length(); i++){
        int mid = i;
        int left = mid, right = mid;
        int count = 0;
        
        if(s[mid-1] == s[mid]) left--;
        else continue;
        
        while(left >= 0 && right < s.length()){            
            if(s[left] != s[right]) break;
            count += 2;
            left--;
            right++;
        }
        if(count > answer) answer = count;
    }
    
    
    if(answer == 0) answer++;
    return answer;
}
```

* `투포인터`를 활용하여 해결
* 문자열의 중앙 부분을 이동시키면서 포인터를 양쪽으로 확장해 나감
* 문자열의 길이가 짝수인 경우, 홀수인 경우 팰린드롬 판단이 달라지므로 두 경우 다 검사해야 함

### [모두 0으로 만들기](https://school.programmers.co.kr/learn/courses/30/lessons/76503)

```cpp
#include <string>
#include <vector>
#include <queue>
#include <algorithm>
#include <cmath>

using namespace std;

long long solution(vector<int> a, vector<vector<int>> edges) {
    long long answer = 0;
    long long sum = 0;
    
    // 가능 판단
    for(const int& weight : a)
        sum += weight;
    if(sum != 0) return -1;
    
    // 각 노드의 연결 정보 저장
    vector<vector<int>> graph(a.size());
    for(const vector<int>& edge : edges){
        graph[edge[0]].push_back(edge[1]);
        graph[edge[1]].push_back(edge[0]);
    }
    
    // 트리의 루트 노드를 찾음
    int root;
    for(int i=0; i<graph.size(); i++){
        if(graph[i].size() == 1) {
            root = i;
            break;
        }
    }
    
    // BFS로 방문 순서를 찾음
    queue<int> q;
    vector<bool> visit(graph.size(), false);    
    vector<int> sequence(graph.size(), -1);     // 방문 순서
    vector<int> parents(graph.size(), -1);      // 각 노드의 부모
    int count = 0;
    q.push(root);
    visit[root] = true;
    sequence[count++] = root;
    
    while(!q.empty()){
        int current = q.front();
        q.pop();
        for(const int& next : graph[current]){
            if(!visit[next]){
                q.push(next);
                visit[next] = true;
                parents[next] = current;
                sequence[count++] = next;
            }
        }
    }
    
    // 방문 순서를 역순으로 해 자식에서 부모로 올라감
    reverse(sequence.begin(), sequence.end());
    vector<long long> weights(a.size());
    for(int i=0; i<a.size(); i++) weights[i] = a[i];
    
    for(const int& current : sequence){
        int parent = parents[current];
        weights[parent] += weights[current];
        answer += abs(weights[current]);
        weights[current] = 0; 
    }
        
    return answer;
}
```

* 문제에서 제시되는 그래프는 트리 구조이므로 
* 자식 노드에서 루트 노드까지 올라가면서 자식의 weight를 0으로 변경하면 모든 노드가 0이 되고 이게 최소 횟수임
* 그래프를 탐색하여 트리를 구성한 후 **BFS**를 통해 트리를 탐색하면 마지막에 도착한 노드들이 리프 노드들임
* 방문 순서를 역순으로 방문해 weight를 0으로 만들면서 루트까지 올라가면 정답

### [디스크 컨트롤러](https://school.programmers.co.kr/learn/courses/30/lessons/42627)

```cpp
#include <string>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

auto cmp = [](const vector<int>& a, const vector<int>& b){
        if(a[2] != b[2]) return a[2] > b[2];
        else if(a[1] != b[1]) return a[1] > b[1];
        else return a[0] > b[0];
    };

int solution(vector<vector<int>> jobs) {
    int answer = 0;
    
    // 작업을 요청시간 순으로 정렬
    sort(jobs.begin(), jobs.end(), [](const vector<int>& a, const vector<int>& b){
        return a[0] < b[0];
    });
    
    // 문제의 정렬 순서대로 정렬하는 우선순위 큐
    priority_queue<vector<int>, vector<vector<int>>, decltype(cmp)> pq(cmp);
    
    int time = 0;
    int count = 0;
    int cur = 0;
    while(count != jobs.size()){ 
        // 현재 시간까지의 작업 요청을 큐에 넣음
        while(cur < jobs.size() && jobs[cur][0] <= time){
            pq.push({cur, jobs[cur][0], jobs[cur][1]});
            cur++;
        }
        
        // 큐의 가장 앞에 있는 작업을 진행
        if(!pq.empty()){
            vector<int> current = pq.top();
            pq.pop();
            count++;
            time += current[2];
            answer += time - current[1];
        }
        else time = jobs[cur][0];   // 큐가 비어있으면 다음 작업의 요청 시간으로 이동
    }
    return answer/jobs.size();
}
```

* 문제의 조건대로 작업을 저장하는 **우선순위큐**를 추가
* 큐 안에 있는 작업을 앞에서부터 꺼내 처리
* 큐에 작업을 넣을 때, 현재 시간에 요청이 된 작업만 넣어야 하므로 현재 시간을 기준으로 작업을 넣음

## 3. 우선순위 큐 비교 함수

* C++에서 우선순위큐의 비교 함수를 직접 구현해서 사용할 일이 그동안 없었음
* [디스크 컨트롤러](https://school.programmers.co.kr/learn/courses/30/lessons/42627) 문제를 풀며 `sort()` 에서 비교 함수를 넣듯이 괄호 안에 **람다 함수**로 넣었지만 **컴파일 에러**
* `sort()`는 함수고 `priority_queue<T>`는 클래스 템플릿이기에 비교 규칙을 **타입**으로 정의해야 함
* 우선순위 큐에 비교 함수를 지정하는 방법은 주로 세가지를 사용

### 구조체 + bool operator() 오버로딩
```cpp
// 구조체
struct Compare {
    // bool operator()
    bool operator()(const vector<int>& a, const vector<int>& b) {
        return a[1] > b[1]; 
    }
};

// 선언
priority_queue<vector<int>, vector<vector<int>>, Compare> pq;
```

### 구조체 + 연산자 오버로딩
```cpp
// 구조체
struct Node {
    int val, priority;
    // < 연산자를 오버로딩
    bool operator<(const Node& other) const {
        return priority > other.priority;
    }
};

// 별도의 비교 클래스 없이 자료형만 선언
priority_queue<Node> pq;
```

### 람다식
```cpp
// 람다식
auto cmp = [](const vector<int>& a, const vector<int>& b) {
    return a[1] > b[1];
};

// 타입은 decltype, 실제 비교 객체는 생성자로 전달
priority_queue<vector<int>, vector<vector<int>>, decltype(cmp)> pq(cmp);
```
