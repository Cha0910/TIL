### [롤케이크 자르기](https://school.programmers.co.kr/learn/courses/30/lessons/132265)

```cpp
#include <string>
#include <vector>

using namespace std;

int solution(vector<int> toppings) {
    int answer = 0;

    // 왼쪽, 오른쪽이 갖고 있는 토핑의 종류의 개수
    int leftCount = 0, rightCount = 0;  

    // 왼쪽, 오른쪽이 갖고있는 각 종류의 토핑의 빈도 수
    vector<int> leftTopping(10001, 0), rightTopping(10001, 0);  
    
    // 모든 토핑을 오른쪽으로 줌
    for(const int& topping : toppings){
        if(rightTopping[topping] == 0)
            rightCount++;
        rightTopping[topping]++;
    }
    
    // 토핑을 하나씩 왼쪽으로 줌
    for(const int& topping : toppings){
        if(leftTopping[topping] == 0)   // 처음 넘어온 토핑이면 종류의 개수를 증가
            leftCount++;
        leftTopping[topping]++;
        
        rightTopping[topping]--;    
        if(rightTopping[topping] == 0)  // 더 이상 오른쪽이 해당 토핑이 없으면 종류의 개수를 감소
            rightCount--;
        
        if(leftCount == rightCount) answer++;   // 같아지는 경우의 수 증가
    } 
        
    
    return answer;
}
```

* `슬라이딩 윈도우?`, `누적합?` 기반으로 해결
* 케이크를 자르는 지점을 점점 오른쪽으로 옮기며 조건을 확인
* 새로운 토핑이 추가, 제거되었을 때 해당 토핑으로 왼쪽과 오른쪽의 토핑 종류의 수가 변경되는 경우를 확인

### [연속 펄스 부분 수열의 합](https://school.programmers.co.kr/learn/courses/30/lessons/161988)

```cpp
#include <string>
#include <vector>

using namespace std;

long long solution(vector<int> sequence) {
    long long answer = 0;
    vector<long long> dpPositive(sequence.size()), dpNegative(sequence.size());
    dpPositive[0] = sequence[0]; dpNegative[0] = -sequence[0];
    answer = max(dpPositive[0], dpNegative[0]);
    
    for(int i=1; i<sequence.size(); i++){
        dpPositive[i] = max(dpNegative[i-1] + sequence[i], (long long)sequence[i]);
        dpNegative[i] = max(dpPositive[i-1] - sequence[i], (long long)-sequence[i]);
        
        answer = max(answer, dpPositive[i]);
        answer = max(answer, dpNegative[i]);
    }
    
    return answer;
}
```

* `DP` 문제
* 문제에서 **펄스 수열**은 **+1**, **-1**이 번갈아 가면서 나오는 것 
* 즉, 앞의 숫자가 **+1** 이면 현재 숫자는 **-1**, 앞의 숫자가 **-1** 이면 현재 숫자는 **+1** 이라는 뜻
* 각각의 자리가 +1인 경우, -1인 경우 최댓값을 저장해 DP로 계산

### [블록 이동하기](https://school.programmers.co.kr/learn/courses/30/lessons/60063)

```cpp
#include <string>
#include <vector>
#include <queue>
#include <map>

using namespace std;

// 왼쪽 상단에 있는 좌표가 앞으로 옴
vector<int> sortPos(const vector<int>& pos){
    if(pos[0] < pos[2] || (pos[0] == pos[2] && pos[1] < pos[3])) return pos;
    else return {pos[2], pos[3], pos[0], pos[1]};  
}

// 현재 위치에서 회전 가능한 위치를 벡터에 담아 반환
vector<vector<int>> GetRotatePos(const vector<vector<int>>& board, const vector<int>& currentPos){
    vector<vector<int>> RotatePos;
    bool isVertical = currentPos[1] == currentPos[3] ? true : false;    
    int d[2] = {-1, 1};
    
    // currentPos[0], currentPos[1]을 회전축으로
    for(int i=0; i<2; i++){
        int rotatePosY = isVertical ? currentPos[0] : currentPos[0] + d[i];
        int rotatePosX = !isVertical ? currentPos[1] : currentPos[1] + d[i];
        int checkY = isVertical ? currentPos[0]+1 : currentPos[0]+d[i];
        int checkX = isVertical ? currentPos[1]+d[i] : currentPos[1]+1;
        
        if(rotatePosX < 0 || rotatePosX >= board.size() || rotatePosY < 0 || rotatePosY >= board.size()
          || checkX < 0 || checkX >= board.size() || checkY < 0 || checkY >= board.size())
            continue;
        
        if(!board[rotatePosY][rotatePosX] && !board[checkY][checkX])
            RotatePos.push_back(sortPos({currentPos[0], currentPos[1], rotatePosY, rotatePosX}));
    }
    
    // currentPos[2], currentPos[3]을 회전축으로
    for(int i=0; i<2; i++){
        int rotatePosY = isVertical ? currentPos[2] : currentPos[2] + d[i];
        int rotatePosX = !isVertical ? currentPos[3] : currentPos[3] + d[i];
        int checkY = isVertical ? currentPos[2]-1 : currentPos[2]+d[i];
        int checkX = isVertical ? currentPos[3]+d[i] : currentPos[3]-1;
        
        if(rotatePosX < 0 || rotatePosX >= board.size() || rotatePosY < 0 || rotatePosY >= board.size()
          || checkX < 0 || checkX >= board.size() || checkY < 0 || checkY >= board.size())
            continue;
        
        if(!board[rotatePosY][rotatePosX] && !board[checkY][checkX])
            RotatePos.push_back(sortPos({currentPos[2], currentPos[3], rotatePosY, rotatePosX}));
    }
    
    return RotatePos;
}

// 현재 위치에서 이동 가능한 위치를 벡터에 담아 반환
vector<vector<int>> GetMovePos(const vector<vector<int>>& board, const vector<int>& currentPos){
    vector<vector<int>> MovePos;
    int dy[4] = {-1, 1, 0, 0};
    int dx[4] = {0, 0, -1, 1};
    
    for(int i=0; i<4; i++){
        int y1 = currentPos[0] + dy[i], x1 = currentPos[1] + dx[i];
        int y2 = currentPos[2] + dy[i], x2 = currentPos[3] + dx[i];
        
        if(x1 < 0 || x1 >= board.size() || y1 < 0 || y1 >= board.size()
          || x2 < 0 || x2 >= board.size() || y2 < 0 || y2 >= board.size())
            continue;
        
        if(!board[y1][x1] && !board[y2][x2])
            MovePos.push_back(sortPos({y1, x1, y2, x2}));
    }
    return MovePos;
}

int solution(vector<vector<int>> board) {
    int answer = 0;
    vector<int> startPos = {0, 0, 0, 1};
    queue<vector<int>> q;
    map<vector<int>, int> visit;
    q.push(startPos);
    visit[startPos] = 0;
    
    while(!q.empty()){
        vector<int> currentPos = q.front();
        q.pop();
        
        if(currentPos[2] == board.size()-1 && currentPos[3] == board.size()-1){
            answer = visit[currentPos];
            break;
        }
        
        vector<vector<int>> rotatePos = GetRotatePos(board, currentPos);
        vector<vector<int>> movePos = GetMovePos(board, currentPos);
        vector<vector<int>> nextPos;
        nextPos.insert(nextPos.end(), rotatePos.begin(), rotatePos.end());
        nextPos.insert(nextPos.end(), movePos.begin(), movePos.end());
        
        for(const vector<int>& next : nextPos){
            if(visit.count(next) == 0){
                visit[next] = visit[currentPos]+1;
                q.push(next);
            }
        }
    }
    
    return answer;
}
```

* `BFS`, `구현` 문제
* 최단 거리를 구하는 것 자체는 다른 문제들과 동일함
* 회전할 때, 회전이 가능한지를 판단하는 것을 구현하는 게 매우 귀찮은 문제
* 드론이 좌푯값을 두 개를 갖기 때문에 `map`으로 방문 체크를 함
