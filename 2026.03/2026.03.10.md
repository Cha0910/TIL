### [숫자 카드 나누기](https://school.programmers.co.kr/learn/courses/30/lessons/62048)

```cpp
#include <string>
#include <vector>
#include <numeric>

using namespace std;

// 벡터의 최대공약수 반횐
int getGCD(const vector<int>& array){
    int result = array[0];
    for(size_t i=1; i<array.size(); i++){
        result = gcd(result, array[i]);
    }
    return result;
}

int solution(vector<int> arrayA, vector<int> arrayB) {
    int answer = 0;
    int gcdA, gcdB;
    gcdA = getGCD(arrayA);  // A의 최대공약수
    gcdB = getGCD(arrayB);  // B의 최대공약수
    
    for(size_t i=0; i<arrayB.size(); i++){
        if(arrayB[i] % gcdA == 0) break;            // A의 최대공약수로 B의 숫자를 하나라도 나누면 정지
        if(i == arrayB.size() - 1) answer = gcdA;   // B의 모든 수를 나누지 못하면 저장
    }
    
    for(size_t i=0; i<arrayA.size(); i++){
        if(answer >= gcdB || arrayA[i] % gcdB == 0) // A의 최대공약수가 더 크고 조건을 만족하거나
            break;                                  // B의 최대공약수로 A의 숫자를 하나라도 나누면 정지
        if(i == arrayA.size() - 1) answer = gcdB;   // A의 모든 수를 나누지 못하면 저장
    }
    
    return answer;
}
```

* 각 집합의 최대공약수가 반대 집합을 나눌 수 있는지 판별
* 최대공약수가 조건을 만족하면 그 아래 수는 검사할 필요가 없음
* 최대공약수로 나눌 수 있는 수가 있으면 그 아래의 약수로도 나눌 수 있으므로 최대공약수로만 판별

### [마법의 엘리베이터](https://school.programmers.co.kr/learn/courses/30/lessons/148653)

```cpp
#include <string>
#include <vector>

using namespace std;

int solution(int storey) {
    int answer = 0;
    while(storey > 0){
        int floor = storey % 10;
        storey /= 10;
        
        if (floor > 5 || (floor == 5 && storey % 10 >= 5)) {
            answer += 10 - floor;
            storey++;
        }
        
        else answer += floor;
    }
        
    return answer;
}
```
* 1의 자릿수가 5일 때 올림 또는 내림을 판별하는 것이 중요한 문제
* 5가 2개 연속으로 있으면 올림, 내림의 결과가 동일함
* 5가 3개 이상 연속으로 있을 경우 올림 하는 것이 최단거리
* 그 외의 경우에는 6 이상이면 올림, 미만이면 내림 하는 것이 최단거리

### [아이템 줍기](https://school.programmers.co.kr/learn/courses/30/lessons/87694)

```cpp
#include <string>
#include <vector>
#include <iostream>
#include <queue>

using namespace std;

int solution(vector<vector<int>> rectangle, int characterX, int characterY, int itemX, int itemY) {
    int answer = 0;
    int maxX = 0, maxY = 0;
    vector<vector<int>> map(102, vector<int>(102, 0));
    
    for(vector<int> rect : rectangle) {
        // 둘레끼리 만나는 것을 방지하기 위해 2 곱함
        int x1 = rect[0] * 2, y1 = rect[1] * 2;
        int x2 = rect[2] * 2, y2 = rect[3] * 2;

        if(x2 > maxX) maxX = x2;
        if(y2 > maxY) maxY = y2;

        for(int i=x1; i<=x2; i++) {
            for(int j=y1; j<=y2; j++) {
                if(i>x1 && i<x2 && j>y1 && j<y2) map[i][j] = -1;    // 사각형 내부는 -1
                else if(map[i][j] != -1) map[i][j] = 1;             // 둘레는 1
            }
        }
    }
    
    // 캐릭터 위치에서 아이템 위치까지 BFS
    queue<pair<int,int>> q;
    vector<vector<int>> visit(maxX+1, vector<int>(maxY+1, 0));
    int dx[4] = {1, -1, 0, 0};
    int dy[4] = {0, 0, 1, -1};  
    
    q.push({characterX*2, characterY*2});
    
    while(!q.empty()){
        pair<int, int> current = q.front();
        q.pop();
        
        for(int i=0; i<4; i++){
            int x = current.first + dx[i];
            int y = current.second + dy[i];
            if(x>=0 && x<maxX+1 && y>=0 && y<maxY+1 && visit[x][y] == 0 && map[x][y] == 1){
                q.push({x, y});
                visit[x][y] = visit[current.first][current.second] + 1;
            }
        }
        if(visit[itemX*2][itemY*2] != 0) break;
    }
    
    answer = visit[itemX*2][itemY*2] / 2; // 2 곱한걸 다시 나눠줌
    return answer;
}
```
* 단순히 둘레에는 1, 내부에는 -1을 넣어 테두리를 따라 BFS를 하면 둘레끼리 만나는 지점에서 잘못된 방향으로 이동하게 됨
* 사각형의 크기를 2배 키워서 둘레끼리 만나는 지점을 지운 후 마지막에 이동 거리를 2로 나누어 줌

### [단속카메라](https://school.programmers.co.kr/learn/courses/30/lessons/42884)
```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;

int solution(vector<vector<int>> routes) {
    int answer = 0;
    
    sort(routes.begin(), routes.end()); // 차량 진입 지점을 기준으로 오름차순
    int cameraStart = -30001;
    int cameraEnd = -30001;

    for(vector<int> route : routes){
        if(cameraEnd < route[0]){   // 현재 카메라의 범위로 새로운 자동차를 단속 못하면
            cameraStart = route[0]; // 새로운 단속 시작 위치
            cameraEnd = route[1];   // 새로운 단속 종료 위치
            answer++;               // 카메라 개수 추가
        }
        else{                       // 현재 카메라의 범위로 새로운 자동차를 단속할 수 있으면
            cameraStart = route[0]; // 단속 시작 위치를 새로운 자동차의 진입 지점으로 변경
            cameraEnd = route[1] < cameraEnd ? route[1] : cameraEnd; // 단속 종료 위치를 더 낮은 값으로 변경
        }
    }

    return answer;
}
```

## 3. 퍼즐 어드벤처 디자인하기

### 베이스 함정 블루프린트
* `fnApplyDamageToTargets` 함수를 추가한 후 `Apply Damage` 노드를 통해 **Box Collision**에 충돌한 **Character** 클래스 액터에게 데미지를 부여
* `EventGraph`에서 `Set Timer by Fucntion Name` 노드와 `TimerHandler`를 통해 액터가 생성될 때 타이머를 설정하고 일시정지 해둠
* `Overlap` 이벤트를 통해 타이머를 On/Off 하여 충돌 중에만 데미지를 부여
* 해당 **베이스 함정 블루프린트**를 부모로 하는 가시 함정과 화염 함정을 **자식 블루프린트**로 생성하여 기능 확장

### 플레이어 HUD 업데이트
* `WBP_PlayerHUD`의 `txtHP`를 변수로 설정한 후 `fnSetHP` 함수를 생성
* 캐릭터의 블루프린트에서 `Event Possessed`에서 `fnSetHP` 실행하여 HP를 전달한 후 `Add to Viewport` 노드로 **HUD**를 화면에 띄움
* `Event AnyDamege`에서 `fnSetHP` 함수를 실행해 동적으로 전달

### 게임 오버 및 리스폰
* `WBP_EliutedScreen`라는 이름의 **사용자 위젯**을 생성해 게임 오버 화면 추가
* `Event AnyDamege`에 체력이 0 이하로 내려가면 `Disable Input` 노드로 사용자의 입력을 막는 로직 추가
* 이후 `WBP_EliutedScreen`를 `Add to Viewport` 노드로 화면에 띄운 후 `Delay`, `Get Current Level Name`, `Open Level(by Name)` 노드를 차례로 실행해 일정 시간 후 현재 레벨을 다시 시작
