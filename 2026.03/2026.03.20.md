### [큰 수 만들기](https://school.programmers.co.kr/learn/courses/30/lessons/42883)

```cpp
#include <string>
#include <vector>

using namespace std;

string solution(string number, int k) {
    string answer = "";
    
    // 숫자를 하나씩 봄
    for(int i=0; i<number.length(); i++){
        
        // 숫자를 제거할 수 있고, 현재 숫자가 가장 뒤의 숫자보다 크면
        while(answer.length() !=0 && k>0 && answer.back() < number[i]){
            answer.pop_back();  // 가장 뒤의 숫자를 지움
            k--;                // 제거할 수 있는 횟수를 차감
        }
        answer += number[i];    // 넣을 수 있는 가장 큰 위치에 추가
    }
    
    // 지울 수 있는 횟수가 남아있으면 뒤의 숫자들을 지움
    if(k>0) 
        answer = answer.substr(0, answer.length()-k);
    
    return answer;
}
```

* 숫자를 하나씩 읽으며 확인하는 **그리디** 방식
* 새로운 숫자가 맨 뒤의 숫자보다 크면 맨 뒤의 숫자를 빼는 것을 반복하여 위치를 찾음
* k가 남는 경우(내림차순 정렬과 같이) 뒤에서 남은 k개만큼 삭제

### [기지국 설치](https://school.programmers.co.kr/learn/courses/30/lessons/12979)

```cpp
#include <vector>
#include <cmath>

using namespace std;

int solution(int n, vector<int> stations, int w)
{
    int answer = 0;
    int left = 0;   // 전파가 닿지 않는 아파트 시작점
    int right = 0;  // 전파가 닿지 않는 아파트 끝점
    
    // 마지막 기지국이 마지막 아파트를 포함하지 못할 수 있으므로
    // 아파트의 범위에 영향을 주지 않는 마지막 기지국 추가
    stations.push_back(n+w+1);  
    
    for(const int& station : stations){
        right = station - w - 1;                        // 기지국의 왼쪽 범위 끝
        answer += ceil((right - left) / (w * 2.0 + 1)); // 전파가 닿지 않는 범위와 필요한 기지국 개수를 계산
        left = station + w;                             // 다음 빈 구간의 시작점 업데이트
    }

    return answer;
}
```

* `C++ 코딩테스트 완전 정복` 수강 중 풀어본 **그리디** 문제
* 전파가 닿지 않는 연속된 아파트가 `n`개 있다고 하면 해당 아파트에 필요한 기지국의 개수는 `n`을 전파 범위인 `w * 2 + 1`을 나눈 값을 올림 한 것과 같음
* 설치된 기지국을 순회하며 전파가 닿지 않는 연속된 아파트의 범위를 계산하고, 해당 범위를 채우는 데 필요한 기지국의 개수를 answer에 더해 정답을 구함

### [메뉴 리뉴얼](https://school.programmers.co.kr/learn/courses/30/lessons/72411)

```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

// 조합
vector<string> combination(string str, int n){
    vector<bool> arr(str.size(), false);
    vector<string> result;
    sort(str.begin(), str.end());
    replace(arr.begin()+n, arr.end(), false, true);

    do{
        string com = "";
        for(int i=0; i<arr.size(); i++)
            if(!arr[i]) com+=str[i];
        result.push_back(com);
    }while(next_permutation(arr.begin(), arr.end()));
        
    return result;
}

vector<string> solution(vector<string> orders, vector<int> course) {
    vector<string> answer;
    map<string, int> count;         // 해당 코스의 주문 횟수
    vector<int> maxCount(11, 0);    // 코스 길이별 최대 주문 횟수
    
    // 모든 주문 순회
    for(const string& order : orders){
        for(const int& size : course){
            if(order.size() < size) break;
            for(const string& com : combination(order, size)){  // 가능한 모든 코스에 대해서
                count[com]++;                                   // 주문 횟수 증가
                
                // 최대 주문 횟수 업데이트
                maxCount[com.length()] = max(maxCount[com.length()], count[com]);
            }
        }
    }
    
    // 모든 코스 요리를 순회
    for(auto [menus, count] : count)
        if(maxCount[menus.length()] == count && count > 1) // 가장 많이 주문된 코스이며, 2번 이상 주문 된 코스면
            answer.push_back(menus);                       
    
    return answer;
}
```

* `next_permutation`을 활용하여 모든 코스의 조합을 구함
* `map`으로 코스의 카운트를 세 가장 많이 주문된 코스를 answer에 저장

### [튜플](https://school.programmers.co.kr/learn/courses/30/lessons/64065)

```cpp
#include <string>
#include <vector>
#include <sstream>
#include <algorithm>
#include <set>

using namespace std;

vector<string> split(const string& str, char c){
    vector<string> result;
    stringstream ss(str);
    string temp;
    while(getline(ss, temp, c)){
        if(temp != "") result.push_back(temp);
    }
    return result;
}

vector<int> splitToInt(const string& str, char c){
    vector<int> result;
    stringstream ss(str);
    string temp;
    while(getline(ss, temp, c)){
        if(temp != "") result.push_back(stoi(temp));
    }
    return result;
}

vector<int> solution(string s) {
    vector<int> answer;
    vector<vector<int>> tuples;
    set<int> numbers;
    
    // 문자열을 vector<int>로 파싱
    for(const string& str : split(s, '}')){
        string temp;
        if(str[0] == '{') temp = split(str, '{')[0];
        else temp = split(str, '{')[1];
        tuples.push_back(splitToInt(temp, ','));
    }
    
    // 숫자의 개수를 기준으로 오름차순 정렬
    sort(tuples.begin(), tuples.end(), [](const vector<int>& a, const vector<int>& b){
        return a.size() < b.size();
    });
    
    // 각 벡터에서 처음 나오는 숫자를 answer에 추가
    for(const vector<int>& tuple : tuples){
        for(const int& number : tuple){
            if(numbers.count(number) < 1){
                numbers.insert(number);
                answer.push_back(number);
            }
        }
    }
    return answer;
}
```

* 주어진 문자열을 **파싱**하여 각각의 벡터로 변경
* 벡터의 **길이가 짧은 순**으로 정렬
* 이전 벡터에는 없었지만, 현재 집합에 새로 생긴 숫자가 튜플의 다음 숫자

## 3. 최소 신장 트리 (Minimum Spanning Tree)
* 사이클이 존재하지 않고, 그래프의 모든 정점을 포함하는 트리를 **신장 트리(Spanning Tree)** 라고 함
* **최소 신장 트리**란 무방향 가중치 그래프에서 **가장 적은 비용**을 갖는 신장트리를 의미함

### 프림 알고리즘 (Prim's Algorithm)
* 시작 정점에서 **정점을 하나씩 확장**해 나가는 알고리즘
* 우선순위큐 사용 시 `O(ElogV)`의 시간복잡도를 가짐

* 동작 방식
    1. 임의의 정점을 시작점으로 선택하여 방문 처리
    2. 현재 방문한 정점의 간선 중 **가중치가 가장 작은 간선**을 선택
    3. 선택한 간선과 연결된 정점이 이미 방문한 정점이면 무시하고 다시 선택
    4. 선택된 정점이 방문하지 않은 정점이면 해당 간선을 MST에 추가
    5. 간선의 개수가 `V-1` 개가 될 때까지 2 ~ 4단계를 반복

* 예제 코드
    ```cpp
    #include <iostream>
    #include <vector>
    #include <queue>

    using namespace std;

    // 가중치, 연결 정점
    typedef pair<int, int> edge;

    void prim(int start, int V, vector<edge> graph[]) {
        vector<bool> visited(V + 1, false);
        priority_queue<edge, vector<edge>, greater<edge>> pq; // 가중치 기준 우선순위큐(오름차순) 

        int total_weight = 0; // MST의 총 가중치 합
        int count = 0;        // 방문한 정점의 수

        // 시작 정점 등록
        pq.push({0, start});

        while (!pq.empty()) {
            int weight = pq.top().first;    // 선택한 간선의 가중치
            int current = pq.top().second;  // 선택한 간선의 목표 정점
            pq.pop();

            // 중복 방문 확인
            if (visited[current]) continue;

            // 방문 처리 및 최소 신장 트리의 비용 증가
            visited[current] = true;
            total_weight  += weight;
            
            cout << "Node: " << current << " Visited (Weight: " << weight << ")" << endl;

            // 모든 정점을 방문했다면 종료
            if (++count == V) break;

            // 인접한 노드 중 방문하지 않은 노드를 간선 후보군에 추가
            for (auto& next_edge : graph[current]) {
                int next_node = next_edge.second;
                int next_weight = next_edge.first;
                
                if (!visited[next_node]) {
                    pq.push({next_weight, next_node});
                }
            }
        }

        cout << "MST Weight: " << total_weight << endl;
    }

    int main() {
        int V = 5; // 정점 개수
        vector<edge> graph[6];

        // 그래프 구성
        graph[1].push_back({2, 2}); graph[1].push_back({3, 3});
        graph[2].push_back({2, 1}); graph[2].push_back({8, 3}); graph[2].push_back({5, 4}); graph[2].push_back({6, 5});
        graph[3].push_back({3, 1}); graph[3].push_back({8, 2}); graph[3].push_back({1, 4});
        graph[4].push_back({5, 2}); graph[4].push_back({1, 3}); graph[4].push_back({4, 5});
        graph[5].push_back({6, 2}); graph[5].push_back({4, 4});

        prim(1, V, graph);

        return 0;
    }

    /* Output:
    Node: 1 Visited (Weight: 0)
    Node: 2 Visited (Weight: 2)
    Node: 3 Visited (Weight: 3)
    Node: 4 Visited (Weight: 1)
    Node: 5 Visited (Weight: 4)
    MST Weight: 10
    */
    ```

### 크루스칼 알고리즘 (Kruskal Algorithm)
* 모든 간선 중 **가중치가 가장 작은 간선부터 선택**해 나가는 알고리즘
* 간선 정렬 시간으로 인해 `O(ElogE)` 또는 `O(ElogV)`의 시간복잡도를 가짐
* 동작 방식
    1. 그래프의 **모든 간선**을 가중치 기준으로 오름차순 정렬함
    2. 정렬된 간선 리스트에서 가중치가 **가장 작은 간선을 선택**
    3. 선택한 간선으로 **사이클**이 발생하는지 확인 (Union & Find)
    4. 사이클이 발생하지 않는다면 해당 간선을 MST에 추가
    5. 선택된 간선의 개수가 `V-1` 개가 될 때까지 2 ~ 4단계를 반복

* 예제 코드
    ```cpp
    #include <iostream>
    #include <vector>
    #include <algorithm>

    using namespace std;

    // 간선 정보 저장
    struct Edge {
        int u, v, weight;
        bool operator<(const Edge& other) const {
            return weight < other.weight;
        }
    };

    // Union-Find를 위한 부모 노드 저장
    int parent[10001];

    // Find 연산: 노드 x의 루트 노드를 찾음
    int Find(int x) {
        if (parent[x] == x) return x;
        return parent[x] = Find(parent[x]);
    }

    // Union 연산: 두 노드가 속한 집합을 합침
    void Union(int a, int b) {
        a = Find(a);
        b = Find(b);
        if (a != b) parent[b] = a;
    }

    void kruskal(int V, vector<Edge>& edges) {
        // 모든 간선을 가중치 기준 오름차순 정렬
        sort(edges.begin(), edges.end());

        // 부모 배열 초기화 (자기 자신을 부모로 설정)
        for (int i = 1; i <= V; i++) parent[i] = i;

        int total_weight = 0;
        int edge_count = 0;

        for (int i = 0; i < edges.size(); i++) {
            int u = edges[i].u;
            int v = edges[i].v;
            int w = edges[i].weight;

            // 사이클 발생 여부 확인 (두 노드의 루트가 다르면 사이클 X)
            if (Find(u) != Find(v)) {
                // MST에 간선 추가 및 두 노드 연결
                Union(u, v);
                total_weight += w;
                edge_count++;
                
                cout << "Edge: " << u << " - " << v << " (Weight: " << w << ") Selected" << endl;

                // 간선의 개수가 V-1개가 되면 종료
                if (edge_count == V - 1) break;
            }
        }

        cout << "MST Weight: " << total_weight << endl;
    }

    int main() {
        int V = 5; // 정점 개수
        vector<Edge> edges;

        // 그래프 데이터 구성: {시작, 끝, 가중치}
        edges.push_back({1, 2, 2}); edges.push_back({1, 3, 3});
        edges.push_back({2, 1, 2}); edges.push_back({2, 3, 8}); edges.push_back({2, 4, 5}); edges.push_back({2, 5, 6});
        edges.push_back({3, 1, 3}); edges.push_back({3, 2, 8}); edges.push_back({3, 4, 1});
        edges.push_back({4, 2, 5}); edges.push_back({4, 3, 1}); edges.push_back({4, 5, 4});
        edges.push_back({5, 2, 6}); edges.push_back({5, 4, 4});

        kruskal(V, edges);

        return 0;
    }

    /* Output:
    Edge: 3 - 4 (Weight: 1) Selected
    Edge: 1 - 2 (Weight: 2) Selected
    Edge: 1 - 3 (Weight: 3) Selected
    Edge: 4 - 5 (Weight: 4) Selected
    MST Weight: 10
    */
    ```

## 4. 피지컬 머티리얼 (Physical Material)
* 피지컬 오브젝트가 월드와 동적으로 **상호작용을 할 때의 반응을 정의**하는 데 사용
* `Floor`에 **피지컬 머티리얼**을 추가하여 바닥의 **표면 타입(Surface Types)** 을 지정해 해당 타입에 알맞은 발소리를 재생함

* 주요 속성
    * Friction (마찰력): 물체가 표면을 지나갈 때의 저항
    * Static Friction (정지 마찰력): 멈춰 있는 물체를 움직이기 시작할 때 필요한 힘
    * Restitution (탄성/반발력): 물체가 충돌했을 때 얼마나 튀어 오르는지를 결정
    * Density (밀도): 물체의 무게를 결정하는 요소

## 5. 애니메이션 노티파이 (Anim Notify)
* 애니메이션 시퀀스 도중 특정 시점에 이벤트를 발생시키는 **트리거**
* 주로 **사운드 재생**, **이펙트 생성**, **공격 판정 On/Off** 등에서 사용
* 특정 프레임에 도달하는 순간 한 번만 이벤트를 실행하는 **Anim Notify**, 구간으로 실행하는 **Anim Notify State**로 나뉨
* **블루프린트** 기반으로 커스텀 노티파이를 생성하여 로직을 부여하거나, **애니메이션 시퀀스** 내에서 생성해 해당 애니메이션의 블루프린트에서 로직을 짤 수 있음
