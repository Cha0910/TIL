### [사라지는 발판](https://school.programmers.co.kr/learn/courses/30/lessons/92345)
```cpp
#include <string>
#include <vector>
#include <iostream>
#include <algorithm>

using namespace std;

class Result{
public:
    bool win; // A 승리: true, B 승리: false
    int count;
    Result(bool _win, int _count) : win(_win), count(_count) {}
};

int dx[4] = {1, -1, 0 ,0};
int dy[4] = {0, 0, 1, -1};

Result dfs(vector<vector<int>> board, vector<int> aloc, vector<int> bloc, bool turn, int count){
    int winCount = 1000, loseCount = -1;
    
    // A의 차례
    if(turn){
        if(board[aloc[0]][aloc[1]] == 0) return {!turn, count}; // 바닥이 없으면 B의 승리 반환
        
        for(int i=0; i<4; i++){
            int x = aloc[0] + dx[i];
            int y = aloc[1] + dy[i];
            
            // 이동 가능하면
            if(x>=0 && x<board.size() && y>=0 && y<board[0].size() && board[x][y] == 1){
                board[aloc[0]][aloc[1]] = 0;
                Result result = dfs(board, {x, y}, bloc, !turn, count+1); // 턴을 1 증가시키고 차례 변경
                if(result.win && result.count < winCount)                 // A가 승리한 경우 중 최소 이동 저장
                    winCount = result.count;
                else if (!result.win && result.count > loseCount)         // A가 패배한 경우 중 최대 이동을 저장
                    loseCount = result.count;
            }
        }
    }
    
    // B의 차례
    else{
        if(board[bloc[0]][bloc[1]] == 0) return {!turn, count}; // 바닥이 없으면 A의 승리 반환
        
        for(int i=0; i<4; i++){
            int x = bloc[0] + dx[i];
            int y = bloc[1] + dy[i];
            
            // 이동 가능하면
            if(x>=0 && x<board.size() && y>=0 && y<board[0].size() && board[x][y] == 1){
                board[bloc[0]][bloc[1]] = 0;
                Result result = dfs(board, aloc, {x, y}, !turn, count+1); // 턴을 1 증가시키고 차례 변경
                if(!result.win && result.count < winCount)                // B가 승리한 경우 중 최소 이동 저장
                    winCount = result.count;
                else if (result.win && result.count > loseCount)          // B가 패배한 경우 중 최대 이동을 저장 
                    loseCount = result.count;
            }
        }
    }
    
    if(winCount != 1000) return {turn, winCount};        // 승리한 경우가 있으면 그 중 가장 낮은 이동 횟수를 반환
    else if (loseCount != -1) return {!turn, loseCount}; // 승리한 경우가 없고 패배한 경우가 있으면 가장 높은 이동 횟수를 반환
    else return {!turn, count};                          // 이동할 수 없는 상황이면 상대방의 승리와 현재 이동 횟수를 반환
}

int solution(vector<vector<int>> board, vector<int> aloc, vector<int> bloc) {
    int answer = -1;
    answer = dfs(board, aloc, bloc, true, 0).count;   // A가 먼저 시작
    return answer;
}
```

* **DFS**와 **Minimax** 알고리즘을 활용하여 해결하는 문제
* DFS와 **최솟값**만 선택하면 승패의 데이터가 없어 단순히 최단 경로가 선택됨
* DFS를 반환할 때 **누가 승리**하는지, **얼마나 이동**했는지에 대한 데이터가 필요함
* 해당 데이터를 기반으로 자신이 승리하는 경우가 있으면 최단 거리, 자신이 패배하는 경우밖에 없으면 최장 거리를 선택

### [배달](https://school.programmers.co.kr/learn/courses/30/lessons/12978)
```cpp
#include <iostream>
#include <vector>
#include <map>
#include <queue>
using namespace std;

// 우선 순위 큐 다익스트라
vector<int> dijkstra(const vector<map<int, int>>& graph, int start, int numTown){
    vector<int> result(numTown+1, 500001);
    result[start] = 0;
    priority_queue<pair<int, int>> pq;
    pq.push({start, 0});
    
    while(!pq.empty()){
        pair<int, int> currentEdge = pq.top();
        pq.pop();
        
        for(const pair<int,int>& edge : graph[currentEdge.first]){
            if(result[currentEdge.first] + edge.second < result[edge.first]){
                result[edge.first] = result[currentEdge.first] + edge.second;
                pq.push(edge);
            }
        }
    }
    
    return result;
}

int solution(int N, vector<vector<int> > road, int K) {
    int answer = 0;
    vector<map<int, int>> graph(N+1);    // graph[a][b] = weight
        
    // 그래프 생성
    for(const vector<int>& data : road){
        int townA = data[0];
        int townB = data[1];
        int weight = data[2];
        
        if(graph[townA].find(townB) == graph[townA].end() || graph[townA][townB] > weight){
            graph[townA][townB] = weight;
            graph[townB][townA] = weight;
        }
    }
    
    // 다익스트라 결과 저장
    vector<int> result = dijkstra(graph, 1, N);
    
    for(const int& weight : result)
        if(weight <= K) answer++;

    return answer;
}
```

* 시작 노드에서 다른 노드까지의 거리를 구하는 문제
* 입력된 벡터로 만든 그래프의 모든 비용이 **양수**이므로 **다익스트라 알고리즘**을 사용하면 됨
* 다익스트라로 나온 결과를 순회하며 `K`비용 안으로 도달 가능한 곳이면 `answer`에 1을 더함

### [1,2,3 떨어트리기](https://school.programmers.co.kr/learn/courses/30/lessons/150364)
```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

class Node{
private:
    vector<Node*> childs;
    int num;
    int sum;
    int selectedEdge;
    
public:
    Node(int _num) : num(_num), sum(0), selectedEdge(0) {}
    
    ~Node() {
        for(Node* child : childs)
            delete child;
    }
    
    void addChild(Node* node){
        childs.push_back(node);
    }
    
    void changeSelecetdEdge(){
        selectedEdge++;
        selectedEdge %= childs.size();
    }
    
    int inputNumber(int number){
        int inputNode;
        if(childs.size() != 0){
            inputNode = childs[selectedEdge]->inputNumber(number);
            changeSelecetdEdge();
        }
        else{
            sum += number;
            inputNode = num;
        }
        return inputNode;
    }
    
    int getSelectedEdge() const{
        return selectedEdge;
    }
};

vector<int> solution(vector<vector<int>> edges, vector<int> target) {
    vector<int> answer;
    
    // edge 정렬
    sort(edges.begin(), edges.end(), [](const vector<int>& a, const vector<int>& b){
        if(a[0] != b[0]) return a[0] < b[0];
        else return a[1] < b[1];
    });
    
    // 마지막 노드의 번호
    int numNodes = target.size();
    
    // 노드 생성
    vector<Node*> nodes(numNodes);
    for(int i = 0; i < numNodes; i++) 
        nodes[i] = new Node(i);
    
    // 트리 연결
    for(const vector<int>& edge : edges){
        int parent = edge[0] - 1;
        int child = edge[1] -1;
        nodes[parent]->addChild(nodes[child]);
    }
    
    // 리프 노드에 숫자가 들어가는 사이클 
    vector<int> cycle;
    bool flag = true;
    while(flag){
        flag = false;
        cycle.push_back(nodes[0]->inputNumber(0));
        
        // 모든 노드의 길이 처음 노드와 동일하면 한바퀴를 돈 것
        for(const Node* node : nodes){
            if(node->getSelectedEdge() != 0) flag = true;
        }  
    }
    
    // 각 리프노드에 숫자가 들어가는 횟수를 저장
    int currentNodeIndex = 0;
    int ballCount = 0;
    vector<int> count(numNodes);
    while(true){
        bool flag = true;
        int currentNode = cycle[currentNodeIndex];
        count[currentNode]++;
        ballCount++;
        
        for(int i=0; i<numNodes; i++){
            if(target[i] > count[i]*3){     // 아직 숫자를 더 넣어야 하는 리프 노드가 존재하면
                flag = false;               // 다음 리프 노드로 이동
                break;                     
            }               
            else if(count[i] > target[i])   // 숫자를 더 못넣는 리프 노드가 생기면 불가능한 것
                return {-1};  
        }
        
        if(flag) break; // 모든 리프노드가 target의 숫자를 만들 수 있으면 정지
        else currentNodeIndex = (currentNodeIndex+1) % cycle.size();
    }
    
    // 각 리프노드에 숫자가 들어가는 순서를 오름차순으로 꺼내기 위해 큐에 저장
    vector<queue<int>> balls(numNodes);
    for(int i=0; i<balls.size(); i++){
        if(count[i] == 0) continue;
        
        vector<int> result(count[i], 1);    // 들어가는 숫자의 개수
        target[i] -= count[i];              // 남은 더해야할 수
        
        int cur = result.size()-1;      // 가장 오른쪽에 있는 숫자
        while(target[i] > 0){           // target이 0이 될 때까지
            result[cur]++;              // 1을 더함
            target[i]--;                
            if(result[cur] == 3) cur--; // 숫자가 3 이상이면 더 더할 수 없으므로 커서를 옮김
        }
        for(const int& num :result)     // 큐에 오름차 순으로 삽입
            balls[i].push(num);
    }
    
    // 숫자를 넣은 순서대로 큐에서 꺼낸 후 answer에 저장
    for(int i=0; i<ballCount; i++){
        int currentNode = cycle[i % cycle.size()];
        answer.push_back(balls[currentNode].front());
        balls[currentNode].pop();
    }
    
    delete nodes[0];
    return answer;
}
```

* 숫자가 들어가는 리프노드의 **사이클**을 찾고 각 리프노드에 숫자가 **몇 번 들어가는지**를 찾는 문제
* 트리 구성이랑 사이클 찾기까지는 금방 했는데 사이클을 어떻게 활용할지 생각해 내는 데 오래걸림

## 3. 계수 정렬(Counting Sort)
* 원소들간의 비교를 하지 않고 숫자의 **카운팅**으로 정렬하는 알고리즘
* `O(n+K)`의 시간복잡도를 가지지만 최솟값, 최댓값의 **범위가 커질수록** 메모리 낭비가 심해지며 매우 느려짐

### 동작 방식
1. 데이터의 최댓값을 확인한 뒤 최댓값의 크기만큼 **리스트**를 생성함
2. 데이터를 하나씩 읽으며 해당 숫자의 **인덱스 값을 1 증가**시킴
3. 안정 정렬을 위해 생성한 리스트를 **누적합**으로 변환
4. 원본 데이터를 **뒤에서** 부터 읽으며 재배치

### 예제 코드
```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void CountingSort(vector<int>& array) {
    int max = *max_element(array.begin(), array.end()); // 최댓값 저장
    vector<int> count(max + 1, 0);
    vector<int> result(array.size());

    // 1. 빈도수 계산
    for (int num : array) count[num]++;

    // 2. 누적 합 계산
    for (int i = 1; i <= max; i++) count[i] += count[i - 1];

    // 3. 재배치
    for (int i = array.size()-1; i >= 0; i--) {
        result[count[array[i]] - 1] = array[i]; // 누적합을 인덱스로 변환하여 저장
        count[array[i]]--; // 
    }
    array = result;
}

int main() {
    vector<int> input = {4, 2, 2, 8, 3, 3, 1};
    CountingSort(input);
    for(int num : input)
        cout << num << " "; // Output: 1 2 2 3 3 4 8 

    return 0;
}
```

**1. 빈도수 계산**

| 인덱스 | 0 | 1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 |
| --- | --- | --- | --- | --- | --- | --- | --- | --- | --- |
| **개수** | 0 | 1 | 2 | 2 | 1 | 0 | 0 | 0 | 1 |

**2. 누적 합 계산**
| 인덱스 | 0 | 1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 |
| --- | --- | --- | --- | --- | --- | --- | --- | --- | --- |
| **누적값** | 0 | 1 | 3 | 5 | 6 | 6 | 6 | 6 | 7 |

**3. 재배치**
| 단계 | `array`의 역순 | `count` | `result` | `count` 감소 |
| --- | --- | --- | --- | --- |
| **1** | `1` | `count[1]`=1 -> 인덱스 0 | `[1, 0, 0, 0, 0, 0, 0]` | `count[1]` = 0 |
| **2** | `3` | `count[3]`=5 -> 인덱스 4 | `[1, 0, 0, 0, 3, 0, 0]` | `count[3]` = 4 |
| **3** | `3` | `count[3]`=4 -> 인덱스 3 | `[1, 0, 0, 3, 3, 0, 0]` | `count[3]` = 3 |
| **4** | `8` | `count[8]`=7 -> 인덱스 6 | `[1, 0, 0, 3, 3, 0, 8]` | `count[8]` = 6 |
| **5** | `2` | `count[2]`=3 -> 인덱스 2 | `[1, 0, 2, 3, 3, 0, 8]` | `count[2]` = 2 |
| **6** | `2` | `count[2]`=2 -> 인덱스 1 | `[1, 2, 2, 3, 3, 0, 8]` | `count[2]` = 1 |
| **7** | `4` | `count[4]`=6 -> 인덱스 5 | `[1, 2, 2, 3, 3, 4, 8]` | `count[4]` = 5 |

**Result:** `1 2 2 3 3 4 8`

## 4. 언리얼 엔진의 라이팅

### 라이트 유형

* **디렉셔널 라이트 (Directional Light)**
    * 주요 야외 라이트 또는 굉장히 멀거나 거의 무한히 떨어진 거리에서 드리워지는 것으로 보이는 라이트
    * 무한한 거리에서 씬 전체에 **평행한 빛**을 비춤
    * 주로 태양광을 나타냄

* **스카이 라이트(Sky Light)**
    * 씬의 배경을 캡처하여 레벨의 지오메트리에 적용
    * 씬의 배경(하늘, 구름 등)을 캡쳐해 해당 색상과 밝기를 레벨 전체에 **환경광(Ambient Light)** 으로 적용
    * 그림자가 지는 어두운 부분을 배경의 색을 반사하여 자연스럽게 만들어줌

* **포인트 라이트(Point Lights)**
    * 전구처럼 작동하며 단일 포인트에서 모든 방향으로 라이트를 드리움
    * 한 점에서 시작해 **모든 방향(구형)** 으로 빛을 방출함
    * 거리에 따라 빛이 약해지는 **감쇠 반경(Attenuation Radius)** 을 가짐

* **스포트 라이트(Spot Lights)**
    * 단일 포인트에서 원뿔 세트에 의해 제한된 방향으로 라이트를 발산
    * 손전등이나 무대 조명처럼 한 점에서 **원뿔** 형태로 빛을 방출
    * **내부 원뿔각(Inner Cone Angle)** 과 **외부 원뿔각(Outer Cone Angle)** 을 통해 밝은 부분을 조절할 수 있음

* **렉트 라이트(Rect Lights)**
    * 사각형 표면에서 한 방향으로 라이트를 발산
    * 빛이 퍼지는 **소스 너비(Source Width)** 와 **높이(Source Height)** 를 가짐
    * 주로 TV 화면, 형광등 등에 주로 사용


### 라이트 모빌리티
언리얼 엔진의 라이팅은 **모빌리티**를 통해 게임 플레이 도중 이동, 변경이 가능한지 여부를 설정할 수 있음

* **스태틱(Static)**
    * 게임플레이 도중 어떤 식으로든 **이동 또는 업데이트되지 않는** 라이트 액터에 사용
    * **라이트매스(Lightmass)** 시스템이 빛의 직사광, 반사광, 그림자를 미리 계산하여 스태틱 메시의 표면에 **텍스처**로 입힘

* **스테이셔너리(Stationary)**
    * 게임플레이 도중 **변경될 수는 있지만 이동되지는 않는** 액터에 사용
    * 간접광과 정적인 오브젝트가 만드는 그림자는 **스태틱 라이트처럼 미리 구워냄**
    * 런타임에 움직이는 **다이내믹 오브젝트**에 대해서는 **실시간으로 그림자를 생성**

* **무버블(Movable)**
    *  게임플레이 도중 **추가되거나, 제거되거나, 이동되어야 하는** 라이트 액터에 사용
    * 라이트맵을 전혀 굽지 않고 모든 빛과 그림자를 **런타임 중 매 프레임 실시간으로 연산**
