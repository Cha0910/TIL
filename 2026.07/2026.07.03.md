### [[3차] 압축](https://school.programmers.co.kr/learn/courses/30/lessons/17684)

```cpp
#include <string>
#include <vector>
#include <unordered_map>

using namespace std;

vector<int> solution(string msg) {
    vector<int> answer;
    unordered_map<string, int> dictionary;
    for(int i=0; i<26; i++){
        string temp = "";
        temp += 'A' + i;
        dictionary[temp] = i+1;
    }
    
    int cur = 0;
    while(cur < msg.length()){
        string word = "", prev;
        word += msg[cur];
        prev = word;
        
        while(dictionary.count(word)){
            cur++;
            prev = word;
            word += msg[cur];
        }
        
        answer.push_back(dictionary[prev]);
        dictionary[word] = dictionary.size()+1;
    }
    
    return answer;
}
```

* **해시맵**, **구현** 문제
* 문자열을 순회하며 이전에 동일한 문자열이 존재했는지 맵을 통해 확인

### [[3차] 파일명 정렬](https://school.programmers.co.kr/learn/courses/30/lessons/17686)

```cpp
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

vector<string> split(const string& str, int i){
    vector<string> result(3, "");
    result[0] = to_string(i);
    int cur = 0;
    
    while(cur < str.length() && !isdigit(str[cur])) 
        result[1] += tolower(str[cur++]);
    
    while(cur < str.length() && result[2].length() < 5 && isdigit(str[cur])) 
        result[2] += str[cur++];
    
    return result;
}

vector<string> solution(vector<string> files) {
    vector<string> answer;
    vector<vector<string>> parsing;
    for(int i=0; i<files.size(); i++)
        parsing.push_back(split(files[i], i));
    
    sort(parsing.begin(), parsing.end(), [](const vector<string>& A, const vector<string>& B){
        if(A[1] != B[1]) return A[1] < B[1];
        else if(stoi(A[2]) != stoi(B[2])) return stoi(A[2]) < stoi(B[2]);   
        else return stoi(A[0]) < stoi(B[0]);
    });
    
    for(int i=0; i<parsing.size(); i++)
        answer.push_back(files[stoi(parsing[i][0])]);
    
    return answer;
}
```

* 문자열 **파싱** 및 **정렬**
* 단순히 문제 조건대로 파싱하고 정렬하면 풀림
