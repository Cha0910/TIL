### [[PCCP 기출문제] 4번 / 수레 움직이기](https://school.programmers.co.kr/learn/courses/30/lessons/250134)

```cpp
#include <string>
#include <vector>

using namespace std;

int dy[4] = {-1, 1, 0, 0};
int dx[4] = {0, 0, -1, 1};

bool check(const vector<int>& curr, const vector<int>& next){
    if(next[0] == next[2] && next[1] == next[3]) return false;
    if(next[0] == curr[2] && next[1] == curr[3] && next[2] == curr[0] && next[3] == curr[1]) return false;
    return true;
}

vector<pair<int, int>> getNext(const vector<vector<int>>& maze, pair<int, int> pos, vector<vector<bool>>& visit){
    vector<pair<int, int>> nexts;
    
    for(int i=0; i<4; i++){
        int ny = pos.first + dy[i];
        int nx = pos.second + dx[i];
        if(ny<0 || ny>=maze.size() || nx<0 || nx>=maze[0].size() ||
           visit[ny][nx] || maze[ny][nx] == 5) continue;
        nexts.push_back({ny, nx});
    }
    return nexts;
}
    
void DFS(const vector<vector<int>>& maze, const vector<int>& pos, const vector<int>& dest,
        vector<vector<bool>>& redVisit, vector<vector<bool>>& blueVisit, int turn, int& answer){
    if(turn >= answer) return;
    
    pair<int, int> redPos = {pos[0], pos[1]}, bluePos = {pos[2], pos[3]};
    vector<pair<int, int>> redNexts, blueNexts;
    
    bool redArrive = pos[0] == dest[0] && pos[1] == dest[1];
    bool blueArrive = pos[2] == dest[2] && pos[3] == dest[3];
    
    if(redArrive && blueArrive){
        answer = answer < turn ? answer : turn;
        return;
    }
    
    if(redArrive) redNexts.push_back(redPos);
    else redNexts = getNext(maze, redPos, redVisit);
    if(blueArrive) blueNexts.push_back(bluePos);
    else blueNexts = getNext(maze, bluePos, blueVisit);
    
    for(const auto& redNext : redNexts){
        for(const auto& blueNext : blueNexts){
            int ry = redNext.first, rx = redNext.second, by = blueNext.first, bx = blueNext.second;
            vector<int> next = {ry, rx, by, bx};
            if(check(pos, next)){
                redVisit[ry][rx] = true; blueVisit[by][bx] = true;
                DFS(maze, next, dest, redVisit, blueVisit, turn + 1, answer);
                redVisit[ry][rx] = false; blueVisit[by][bx] = false;
            }
        }
    }
}

int solution(vector<vector<int>> maze) {
    int answer = 20;
    vector<int> start(4), dest(4);
    vector<vector<bool>> redVisit(maze.size(), vector<bool>(maze[0].size(), false));
    vector<vector<bool>> blueVisit = redVisit;
    
    for(int i=0; i<maze.size(); i++){
        for(int j=0; j<maze[i].size(); j++){
            if(maze[i][j] == 1){
                start[0] = i;
                start[1] = j;
                redVisit[i][j] = true;
            }
            else if(maze[i][j] == 2){
                start[2] = i;
                start[3] = j;
                blueVisit[i][j] = true;
            }
            else if(maze[i][j] == 3){
                dest[0] = i;
                dest[1] = j;
            }
            else if(maze[i][j] == 4){
                dest[2] = i;
                dest[3] = j;
            }
        }
    }
    
    DFS(maze, start, dest, redVisit, blueVisit, 0, answer);
    
    return answer == 20 ? 0 : answer;
}
```

* **백트래킹** 문제
* `maze`의 범위가 `4*4` 밖에 안 돼서 완전 탐색 가능
* 빨간색 수레의 위치, 파란색 수레의 위치를 **하나의 상태로 관리해 DFS**
* 수레를 **우선 이동** 시킨 후, 이동한 위치를 `check`로 판별해 불가능한 경우에는 탐색 중단

### [[PCCP 기출문제] 4번 / 수식 복원하기](https://school.programmers.co.kr/learn/courses/30/lessons/340210)

```cpp
#include <string>
#include <vector>
#include <sstream>
#include <unordered_set>
#include <algorithm>

using namespace std;

vector<string> split(const string& str, char c){
    vector<string> result;
    stringstream ss(str);
    string temp;
    
    while(getline(ss, temp, c))
        result.push_back(temp);
    
    return result;
}

int convertDemical(string str, int n){
    if(str.length() == 1) return str[0] - '0';
    else if(str.length() == 2) return (str[0] - '0') * n + str[1] - '0';
    else return (str[0] - '0') * n * n + (str[1] - '0') * n + str[2] - '0';
}

string convertNBase(int num, int n){
    string result = "";
    while(num >= n){
        result += to_string(num % n);
        num /= n;
    }
    result += to_string(num);
    reverse(result.begin(), result.end());
    return result;
}

vector<string> solution(vector<string> expressions) {
    vector<string> answer;
    vector<bool> base(10, true);
    base[0] = false; base[1] = false;

    int max = 2;
    for(const string& expression : expressions){
        
        for(const char c : expression){
            if(c < '0' || c > '9') continue;
            int num = c - '0' + 1;
            max = max > num ? max : num; 
        }
        
        if(expression.back() == 'X'){
            answer.push_back(expression);
            continue;
        }
        
        vector<string> data = split(expression, ' ');
        for(int i=max; i<10; i++){
            if(!base[i]) continue;
            int A = convertDemical(data[0], i), B = convertDemical(data[2], i);
            int C = convertDemical(data[4], i);
            if(data[1] == "+" && A+B != C) base[i] = false;
            else if (data[1] == "-" && A-B != C) base[i] = false;
        }
    }
    
    for(int i=2; i<max; i++) base[i] = false;

    for(string& expression : answer){
        unordered_set<string> candidates;
        vector<string> data = split(expression, ' ');
        for(int i=max; i<10; i++){
            if(!base[i]) continue;
            int A = convertDemical(data[0], i), B = convertDemical(data[2], i);
            if(data[1] == "+") candidates.insert(convertNBase(A+B, i));
            else if (data[1] == "-") candidates.insert(convertNBase(A-B, i));
        }
        
        if(candidates.size() > 1) expression.back() = '?';
        else if(!candidates.empty()){
            expression.pop_back();
            expression += *candidates.begin();
        }
    }
    return answer;
}
```

* **완전 탐색**, **진법 변환** 문제
* 식이 보이는 수식에 대하여 **성립하는 진법을 모두 확인**
* 이후 살아남은 진법으로 가려진 수식을 계산
* 이때, 값이 여러 개일 경우 **?**, 하나일 경우 그 값을 입력

### [Range Sum Query - Immutable](https://leetcode.com/problems/range-sum-query-immutable/description/)

```cpp
class NumArray {
public:
    vector<int> pref;

    NumArray(vector<int>& nums) {
        pref.push_back(0);
        for(int i=1; i<nums.size()+1; i++)
            pref.push_back(pref[i-1] + nums[i-1]);
    }
    
    int sumRange(int left, int right) {
        int result = pref[right+1] - pref[left];
        return result;
    }
};
```

* **누적 합** 문제

### [Subarray Sum Equals K](https://leetcode.com/problems/subarray-sum-equals-k/description/)

```cpp
#include <unordered_map>

class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int answer = 0;
        vector<int> pref(nums.size()+1, 0);
        for(int i=1; i<pref.size(); i++)
            pref[i] += pref[i-1] + nums[i-1];

        unordered_map<int, int> count;
        for(int i=0; i<pref.size(); i++){
            answer += count[pref[i]-k];
            count[pref[i]]++;
        }
        
        return answer;
    }
};
```

* **누적 합** 문제
* **이전에 등장한 누적 합** 값을 해시맵에 저장해 **O(N)** 시간 복잡도로 해결 가능
* `i`번째 누적 합이 `A`라 하면, 이전 누적 합 중 `A-k` 가 존재 했는지 확인
* 만약 누적 합이 `A-k`인 경우가 있다면, 그 **누적 합의 뒤부터 `i`번째까지 부분 집합의 합은 k**가 됨
