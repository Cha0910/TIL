## 2. 프로그래머스 문제 풀이

### [키패드 누르기](https://school.programmers.co.kr/learn/courses/30/lessons/67256)

```cpp
#include <string>
#include <vector>
#include <cmath>
using namespace std;
pair<int, int> getPos(int n){
    if(n==0) return {3, 1};
    n--;
    return {n/3 , n%3};
}

int getDist(pair<int, int> s, int d){
    pair<int, int> dest = getPos(d);
    return abs(s.first - dest.first) + abs(s.second - dest.second);
}

string solution(vector<int> numbers, string hand) {
    string answer = "";
    pair<int, int> left = {3, 0}, right = {3, 2};
    
    for(const int num : numbers){
        if(num%3 == 1){
            answer += "L";
            left = getPos(num);
        }
        else if(num != 0 && num%3 == 0){
            answer += "R";
            right = getPos(num);
        }
        
        else{
            int l = getDist(left, num), r = getDist(right, num);
            if(l == r){
                if(hand == "right") {
                    answer += "R";
                    right = getPos(num);
                }
                else {
                    answer += "L";
                    left = getPos(num);
                }
            }
            else if(l < r){
                answer += "L";
                left = getPos(num);
            }
            else{
                answer += "R";
                right = getPos(num);
            }
        }
    }
        
    return answer;
}
```
