### [Number of Provinces](https://leetcode.com/problems/number-of-provinces/description/)

```cpp
#include <queue>

class Solution {
public:
    int findCircleNum(vector<vector<int>>& isConnected) {
        int n = isConnected.size();
        int answer = 0;
        vector<bool> visit(n, false);
        
        for(int i=0; i<n; i++){
            if(visit[i]) continue;
            queue<int> q;
            q.push(i);
            visit[i] = true;
            answer++;

            while(!q.empty()){
                int curr = q.front();
                q.pop();
                for(int j=0;j<n; j++){
                    if(visit[j] || !isConnected[curr][j]) continue;
                    q.push(j);
                    visit[j] = true;
                }
            }
        }

        return answer;
    }
};
```

### [Rotting Oranges](https://leetcode.com/problems/rotting-oranges/description/)

```cpp
class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        int answer = 0;
        vector<vector<int>> visit(m, vector<int>(n, -1));
        queue<pair<int, int>> q;

        int dy[4] = {-1, 1, 0, 0};
        int dx[4] = {0, 0, -1, 1};

        for(int i=0; i<m; i++){
            for(int j=0; j<n; j++){
                if(grid[i][j] == 2){
                    q.push({i, j});
                    visit[i][j] = 0;
                }
            }  
        }

        while(!q.empty()){
            pair<int, int> curr = q.front();
            q.pop();
            int y = curr.first, x = curr.second;
            for(int i=0; i<4; i++){
                int ny = y + dy[i], nx = x + dx[i];
                if(ny<0 || ny>=m || nx<0 || nx>=n || visit[ny][nx] != -1 || grid[ny][nx] != 1) continue;
                q.push({ny, nx});
                visit[ny][nx] = visit[y][x] + 1;
                grid[ny][nx] = 2;
                answer = max(answer, visit[ny][nx]);
            }
        }

        for(int i=0; i<m; i++)
            for(int j=0; j<n; j++)
                if(grid[i][j] == 1)
                    return -1;
                
        return answer;
    }
};
```

* **BFS** 문제
* **시작점이 여러 개**라는 특징이 존재
