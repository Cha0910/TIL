### [요격 시스템](https://school.programmers.co.kr/learn/courses/30/lessons/181188)

```cpp
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int solution(vector<vector<int>> targets) {
    int answer = 0;
    sort(targets.begin(), targets.end(), [](const vector<int>& A, const vector<int>& B){
        return A[0] > B[0];
    });
    for(int i=0; i<targets.size(); i++){
        int turret = targets[i][0];
        answer++;
        for(int j=i+1; j<targets.size(); j++){
            int end = targets[j][1];
            if(turret >= end) break;
            else i++;
        }
    }
    return answer;
}
```

### [광물 캐기](https://school.programmers.co.kr/learn/courses/30/lessons/172927)

```cpp
#include <string>
#include <vector>

using namespace std;
void DFS(vector<int>& picks, const vector<string>& minerals, int fatigue, int start, int& answer){
    if(start >= minerals.size() || (picks[0] == 0 && picks[1] == 0 && picks[2] == 0)){
        answer = min(answer, fatigue);
        return;
    }
        
    for(int i=0; i<picks.size(); i++){
        if(picks[i] == 0) continue;
        
        int sum = 0;
        for(int j=start; j<start+5; j++){
            if(j>= minerals.size()) break;
            if(minerals[j] == "diamond"){
                if(i==0) sum++;
                else if(i==1) sum+=5;
                else sum+=25;
            }
            else if(minerals[j] == "iron"){
                if(i==2) sum+=5;
                else sum++;
            }
            else sum++;
        }
        
        picks[i]--;
        DFS(picks, minerals, fatigue+sum, start+5, answer);
        picks[i]++;
    }
}
int solution(vector<int> picks, vector<string> minerals) {
    int answer = 1251;
    DFS(picks, minerals, 0, 0, answer);
    
    return answer;
}
```
