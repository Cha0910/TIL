### [JadenCase 문자열 만들기](https://school.programmers.co.kr/learn/courses/30/lessons/12951)

```cpp
#include <string>
#include <vector>

using namespace std;

string solution(string s) {
    s[0] = toupper(s[0]);
    for(int i=1; i<s.length(); i++)
        s[i] = s[i-1] == ' ' ? toupper(s[i]) : tolower(s[i]);
    return s;
}
```

### [[1차] 셔틀버스](https://school.programmers.co.kr/learn/courses/30/lessons/17678)

```cpp
#include <string>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;

// 숫자 시간을 문자열로
string intToStr(int time){
    string h = time/60 < 10 ? "0" + to_string(time/60) : to_string(time/60);
    string m = time%60 < 10 ? "0" + to_string(time%60) : to_string(time%60);
    return h + ":" + m;
}

string solution(int n, int t, int m, vector<string> timetable) {
    string answer = "";

    // 문자열 시간을 숫자로
    vector<int> times; times.reserve(timetable.size());
    for(const string& time : timetable){
        int h = stoi(time.substr(0, 2));
        int m = stoi(time.substr(3, 2));
        times.push_back(h*60 + m);
    }
    
    // 크루들의 도착시간을 큐에 저장
    queue<int> q;
    for(const int time : times) q.push(time);

    // 버스의 도착 시간
    int time = 540;

    for(int i=0; i<n; i++){
        int count = 0;
        int curr;

        // i+1번째 버스에 탈 수 있는 사람을 태움
        while(!q.empty() && count < m){
            curr = q.front();
            if(curr > time) break;
            q.pop();
            count++;
        }
        
        // 마지막 버스라면
        if(i==n-1){
            if(count < m) answer = intToStr(time);  // 버스에 여분 좌석이 있으면
            else answer = intToStr(curr-1);         // 버스가 꽉차면 
        }
        time += t;  // 다음 버스 도착 시간으로 변경
    }
    
    return answer;
}
```

* **문자열**, **큐** 문제
* 크루를 정렬한 후 큐에 저장해 버스에 순차적으로 태움
* 마지막 버스에 여분 좌석이 있다면, **마지막 버스가 오는 시간**에 출근하면 탑승 가능
* 여분 좌석이 없으면, **마지막 사람보다 1분 먼저** 출근하면 탑승 가능

### [비밀 코드 해독](https://school.programmers.co.kr/learn/courses/30/lessons/388352)

```cpp
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

// n개의 숫자 중 5개를 고르는 모든 조합을 저장
void combination(int& n, vector<vector<int>>& result, vector<int>& temp, int start){
    if(temp.size() == 5){
        result.push_back(temp);
        return;
    }
    
    for(int i=start; i<=n; i++){
        temp.push_back(i);
        combination(n, result, temp, i+1);
        temp.pop_back();
    }
}

// 현재 시도가 모든 q의 결과와 동일한지
bool check(vector<int>& vec, vector<vector<int>>& q, vector<int>& ans){
    for(int i=0; i<q.size(); i++){
        int count = 0;
        for(int j=0; j<vec.size(); j++)
            if(find(q[i].begin(), q[i].end(), vec[j]) != q[i].end()) count++;
        if(ans[i] != count) return false;
    }
    return true;
}

int solution(int n, vector<vector<int>> q, vector<int> ans) {
    int answer = 0;
    vector<vector<int>> result;
    vector<int> temp;
    result.reserve(150000);
    temp.reserve(5);
    combination(n, result, temp, 1);
    
    for(int i=0; i<result.size(); i++)
        if(check(result[i], q, ans)) answer++;
    
    return answer;
}
```

* **조합**, **완전 탐색**을 활용해 풀이
* 최대 `30`개의 숫자 중 `5`개의 숫자를 고르기에, 조합의 수는 최대 $_{30}C_5 = 142,506$개
* `q`의 길이도 최대 `10`개이므로 완전 탐색으로 풀어도 충분함

## 3. std::find()와 std::binary_search()

* `<algorithm>` 헤더에 포함되어 있는 탐색 알고리즘

* 두 함수 모두 특정 범위내에서 어떤 **값을 찾을 때** 사용하는 함수지만, **작동 방식이나 결과가 다름**

### std::find()

* 특정 범위 내의 원하는 값을 **선형 탐색**으로 찾는 함수

* 입력 인자로 시작점의 **이터레이터**`(first)`와 종료 지점의 **이터레이터**`(last)`를 받으며, `[first, last)` 범위를 탐색

* 가장 먼저 발견한 위치의 **이터레이터**를 반환하며, 탐색 실패 시 `last` **이터레이터**를 반환함

* **선형 탐색**으로 모든 데이터를 방문하기에 $O(N)$의 시간복잡도를 가지지만, 데이터의 **상태와 관계없이** 사용 가능

* **예제 코드**
    ```cpp
    #include <vector>
    #include <algorithm>
    #include <iostream>

    int main(){
        std::vector<int> vec = {5, 8, 6, 7, 4, 2, 9};  // 정렬되지 않은 벡터
        auto iter1 = std::find(vec.begin(), vec.end(), 2);
        auto iter2 = std::find(vec.begin(), vec.end(), 1);

        // 탐색 성공 시 해당 값의 이터레이터를 반환
        if(iter1 != vec.end()){
            int index = std::distance(vec.begin(), iter1);
            std::cout << "vec[" << index << "]: " << *iter1 << std::endl;    // Output: vec[5]: 2
        }

        // 탐색 실패 시 last의 이터레이터를 반환
        if(iter2 == vec.end()){
            std::cout << "vec에 1이 없습니다." << std::endl;    // Output: vec에 1이 없습니다.
        }

        return 0;
    }
    ```

### std::binary_search()

* 특정 범위 내의 원하는 값을 **이진 탐색**으로 찾는 함수

* 입력 인자로 시작점의 **이터레이터**`(first)`와 종료 지점의 **이터레이터**`(last)`를 받으며, `[first, last)` 범위를 탐색

* `std::binary_search()`는 **bool 자료형**을 반환하며 탐색 성공 시 `true`, 실패 시 `false`를 반환

* 탐색 값의 **이터레이터**가 필요한 경우 `std::lower_bound()`를 사용

* **이진 탐색**으로 데이터를 방문하기에 $O(LogN)$의 시간복잡도를 가지지만, 데이터가 **정렬되어 있어야** 사용 가능

* **예제 코드**
    ```cpp
    #include <vector>
    #include <algorithm>
    #include <iostream>

    int main(){
        std::vector<int> vec = {5, 8, 6, 7, 4, 2, 9};  // 정렬되지 않은 벡터
        std::sort(vec.begin(), vec.end());             // 사용 전 정렬 필수, 결과: {2, 4, 5, 6, 7, 8, 9}

        // std::binary_search()는 bool 타입을 반환
        if(std::binary_search(vec.begin(), vec.end(), 2)){
            std::cout << "vec에 2가 있습니다." << std::endl;    // Output: vec 에 2가 있습니다.
        }

        // 이터레이터가 필요하면 lower_bound() 사용
        auto iter1 = std::lower_bound(vec.begin(), vec.end(), 6);
        if(iter1 != vec.end()){
            int index = std::distance(vec.begin(), iter1);
            std::cout << "vec[" << index << "]: " << *iter1 << std::endl;    // Output: vec[3]: 6
        }

        // upper_bound()는 가장 뒤의 이터레이터+1을 반환
        auto iter2 = std::upper_bound(vec.begin(), vec.end(), 7);
        if(iter2 != vec.end()){
            int index = std::distance(vec.begin(), iter2);
            std::cout << "vec[" << index << "]: " << *iter2 << std::endl;    // Output: vec[5]: 8
        }

        return 0;
    }
    ```
